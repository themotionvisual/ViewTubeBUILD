import React, { useState, useEffect } from 'react';

import { Play, Pause, Wand2, Layers, Type, Image as ImageIcon, Video, MousePointer2, Shapes } from 'lucide-react';


// --- 1. THE DATA SCHEMA (The "Source of Truth") ---

const INITIAL_TIMELINE = [

  {

    id: 'track-bg',

    name: 'Background (Video)',

    type: 'video',

    items: [

      { id: 'v1', name: 'Battlefield B-Roll', from: 0, duration: 150, color: 'bg-blue-600' }

    ]

  },

  {

    id: 'track-parallax',

    name: 'Parallax (PNGs)',

    type: 'image',

    items: [

      { id: 'p1', name: 'Napoleon Cutout', from: 30, duration: 90, color: 'bg-green-600' }

    ]

  },

  {

    id: 'track-shapes',

    name: 'Generative Shapes',

    type: 'shape',

    items: [

      { id: 'sh1', name: 'Rock Composition', from: 60, duration: 60, color: 'bg-cyan-500' }

    ]

  },

  {

    id: 'track-overlays',

    name: 'Hooks & Overlays',

    type: 'overlay',

    items: [

      { id: 't1', name: 'Neo-Brutalist Title', from: 10, duration: 60, color: 'bg-yellow-500' }

    ]

  }

];


export default function App() {

  const [tracks, setTracks] = useState(INITIAL_TIMELINE);

  const [frame, setFrame] = useState(0);

  const [isPlaying, setIsPlaying] = useState(false);

  const [prompt, setPrompt] = useState("");

  const [isGenerating, setIsGenerating] = useState(false);

  const MAX_FRAMES = 150; // 5 seconds at 30fps


  // Simulated Player Loop

  useEffect(() => {

    let interval;

    if (isPlaying) {

      interval = setInterval(() => {

        setFrame((prev) => (prev >= MAX_FRAMES ? 0 : prev + 1));

      }, 1000 / 30); // 30 FPS

    }

    return () => clearInterval(interval);

  }, [isPlaying]);


  // Simulated AI Prompt Generation

  const handleAIPrompt = (e) => {

    e.preventDefault();

    if (!prompt) return;

    

    setIsGenerating(true);

    

    setTimeout(() => {

      const newTracks = [...tracks];

      const lowerPrompt = prompt.toLowerCase();

      

      let targetTrackIndex = 3; // Default to overlays

      let newItem = {

        id: `gen-${Date.now()}`,

        from: frame,

        duration: 60,

        name: `Generated: "${prompt.substring(0, 15)}..."`,

        color: 'bg-yellow-400'

      };


      if (lowerPrompt.includes('video') || lowerPrompt.includes('b-roll')) {

        targetTrackIndex = 0; // Background (Video)

        newItem.id = `v-${Date.now()}`;

        newItem.name = `AI B-Roll: ${prompt.split(' ').slice(0,3).join(' ')}`;

        newItem.color = 'bg-red-500';

      } 

      else if (lowerPrompt.includes('parallax') || lowerPrompt.includes('image')) {

        targetTrackIndex = 1; // Parallax (PNGs)

        newItem.id = `p-${Date.now()}`;

        newItem.name = `AI Cutout: ${prompt.split(' ').slice(0,3).join(' ')}`;

        newItem.color = 'bg-green-500';

      }

      else if (lowerPrompt.includes('shape') || lowerPrompt.includes('stack') || lowerPrompt.includes('blob') || lowerPrompt.includes('composition')) {

        targetTrackIndex = 2; // Shapes Track

        newItem.id = `sh-${Date.now()}`; 

        newItem.name = 'AI Shape Stack';

        newItem.color = 'bg-cyan-500';

      }

      else if (lowerPrompt.includes('svg') || lowerPrompt.includes('target')) {

        targetTrackIndex = 3; // Hooks & Overlays (SVG)

        newItem.id = `s-${Date.now()}`; 

        newItem.name = 'AI Target SVG';

        newItem.color = 'bg-purple-500';

      } 

      else {

        targetTrackIndex = 3; // Hooks & Overlays (Text)

        newItem.id = `t-${Date.now()}`; 

        newItem.name = prompt; 

        newItem.color = 'bg-yellow-400';

      }


      newTracks[targetTrackIndex].items.push(newItem);

      setTracks(newTracks);

      setPrompt("");

      setIsGenerating(false);

    }, 800);

  };


  const getVisibleItems = () => {

    const visible = [];

    tracks.forEach(track => {

      track.items.forEach(item => {

        if (frame >= item.from && frame <= item.from + item.duration) {

          visible.push({ ...item, trackType: track.type });

        }

      });

    });

    return visible;

  };


  const visibleItems = getVisibleItems();


  // --- MOTION MATH HELPERS ---

  const calculateSpring = (localFrame, duration = 15) => {

    if (localFrame < 0) return 0;

    if (localFrame >= duration) return 1;

    const x = localFrame / duration;

    const c1 = 1.70158;

    const c3 = c1 + 1;

    return 1 + c3 * Math.pow(x - 1, 3) + c1 * Math.pow(x - 1, 2);

  };


  const interpolate = (val, inMin, inMax, outMin, outMax) => {

    if (val <= inMin) return outMin;

    if (val >= inMax) return outMax;

    return outMin + ((val - inMin) / (inMax - inMin)) * (outMax - outMin);

  };


  const getHexColor = (colorStr) => {

    if (colorStr.includes('yellow')) return '#eab308';

    if (colorStr.includes('purple')) return '#a855f7';

    if (colorStr.includes('blue')) return '#3b82f6';

    if (colorStr.includes('green')) return '#22c55e';

    if (colorStr.includes('red')) return '#ef4444';

    if (colorStr.includes('cyan')) return '#06b6d4';

    return '#ffffff';

  };


  return (

    <div className="flex flex-col h-screen bg-gray-950 text-white font-sans overflow-hidden">

      

      {/* --- TOP BAR --- */}

      <header className="h-14 border-b border-gray-800 flex items-center justify-between px-6 bg-gray-900">

        <div className="flex items-center gap-2">

          <Wand2 className="w-5 h-5 text-yellow-500" />

          <h1 className="font-bold text-lg tracking-tight">CreatorEditor <span className="text-yellow-500">AI</span></h1>

        </div>

        <div className="flex gap-4 text-sm font-medium">

          <button className="px-4 py-1.5 rounded bg-gray-800 hover:bg-gray-700 transition">Export</button>

          <button className="px-4 py-1.5 rounded bg-yellow-500 text-black hover:bg-yellow-400 transition shadow-[2px_2px_0px_white]">Render Video</button>

        </div>

      </header>


      {/* --- MAIN WORKSPACE --- */}

      <div className="flex flex-1 overflow-hidden">

        

        {/* LEFT SIDEBAR: Template Library */}

        <aside className="w-64 border-r border-gray-800 bg-gray-900 flex flex-col hidden md:flex">

          <div className="p-4 border-b border-gray-800 font-bold text-xs uppercase tracking-wider text-gray-400">

            Template Library

          </div>

          <div className="p-4 flex flex-col gap-3 overflow-y-auto">

            <TemplateButton icon={<Type size={16} />} title="Neo-Brutalist Hook" color="border-yellow-500" />

            <TemplateButton icon={<Shapes size={16} />} title="Generative Shape Stack" color="border-cyan-500" />

            <TemplateButton icon={<ImageIcon size={16} />} title="3D Parallax PNG" color="border-green-500" />

            <TemplateButton icon={<MousePointer2 size={16} />} title="SVG Target Chase" color="border-purple-500" />

            <TemplateButton icon={<Video size={16} />} title="Cinematic Zoom B-Roll" color="border-red-500" />

          </div>

        </aside>


        {/* CENTER: The Canvas */}

        <main className="flex-1 flex flex-col bg-black relative">

          

          <div className="flex-1 flex items-center justify-center p-8">

            <div className="relative w-full max-w-[320px] aspect-[9/16] bg-gray-900 border border-gray-800 rounded-lg overflow-hidden shadow-2xl ring-1 ring-white/10">

              

              {visibleItems.map(item => {

                const localFrame = frame - item.from;

                const hexColor = getHexColor(item.color);


                return (

                  <div key={item.id} className="absolute inset-0 flex items-center justify-center pointer-events-none overflow-hidden">

                    

                    {/* 1. CINEMATIC B-ROLL */}

                    {item.trackType === 'video' && (

                      <div className="absolute inset-0 flex items-center justify-center bg-gray-950 overflow-hidden">

                        <div 

                          className="absolute inset-0"

                          style={{ 

                            backgroundColor: hexColor, opacity: 0.3,

                            transform: `scale(${1 + (localFrame * 0.002)}) rotate(${localFrame * 0.01}deg)`,

                            backgroundImage: 'radial-gradient(circle at center, transparent 0%, rgba(0,0,0,0.8) 100%)'

                          }} 

                        />

                        <div className="absolute inset-0 bg-[linear-gradient(rgba(255,255,255,0.05)_1px,transparent_1px),linear-gradient(90deg,rgba(255,255,255,0.05)_1px,transparent_1px)] bg-[size:20px_20px]" />

                        {localFrame < 15 && (

                          <div 

                            className="absolute inset-0 bg-orange-500 mix-blend-overlay"

                            style={{ opacity: interpolate(localFrame, 0, 15, 1, 0), filter: 'blur(20px)' }}

                          />

                        )}

                      </div>

                    )}


                    {/* 2. 3D PARALLAX PNG */}

                    {item.trackType === 'image' && (

                      <div className="absolute inset-0 flex items-center justify-center" style={{ perspective: '800px' }}>

                        <div 

                          className="w-48 h-64 border-2 border-dashed rounded flex flex-col items-center justify-center backdrop-blur-md"

                          style={{

                            backgroundColor: `${hexColor}22`, borderColor: hexColor,

                            transform: `translateZ(50px) rotateY(${Math.sin(localFrame / 20) * 15}deg) rotateX(${Math.cos(localFrame / 20) * 10}deg) translateY(${Math.sin(localFrame / 15) * 10}px)`,

                            boxShadow: `${Math.sin(localFrame / 20) * -20}px ${Math.cos(localFrame / 20) * -20}px 30px rgba(0,0,0,0.8)`,

                            opacity: interpolate(localFrame, 0, 10, 0, 1)

                          }}

                        >

                          <ImageIcon size={48} color={hexColor} className="mb-4 opacity-80" />

                          <span style={{ color: hexColor }} className="font-bold text-center px-4 drop-shadow-lg text-lg leading-tight">{item.name}</span>

                        </div>

                      </div>

                    )}


                    {/* 3. GENERATIVE SHAPES (@remotion/shapes style) */}

                    {item.trackType === 'shape' && (

                      <div className="absolute inset-0 flex items-center justify-center">

                        <div className="relative w-64 h-64">

                          

                          {/* Bottom Blob (Base) */}

                          <div 

                            className="absolute top-[50%] left-[10%] w-[180px] h-[100px] border-[5px] border-black bg-[#e0e7ff]"

                            style={{

                              borderRadius: '40% 60% 70% 30% / 40% 50% 60% 50%',

                              boxShadow: '6px 8px 0px black',

                              transform: `scale(${calculateSpring(localFrame, 15)}) translateY(${Math.sin(localFrame / 15) * 4}px)`,

                              opacity: interpolate(localFrame, 0, 5, 0, 1)

                            }}

                          />


                          {/* Middle Blob */}

                          <div 

                            className="absolute top-[35%] left-[15%] w-[140px] h-[80px] border-[5px] border-black bg-[#93c5fd]"

                            style={{

                              borderRadius: '60% 40% 50% 50% / 50% 40% 60% 50%',

                              boxShadow: '5px 6px 0px black',

                              // Delays entrance by 5 frames

                              transform: `scale(${calculateSpring(localFrame - 5, 15)}) translateY(${Math.sin(localFrame / 12) * -5}px)`,

                              opacity: interpolate(localFrame - 5, 0, 5, 0, 1)

                            }}

                          />


                          {/* Top Blob */}

                          <div 

                            className="absolute top-[18%] left-[20%] w-[90px] h-[80px] border-[5px] border-black bg-[#2563eb]"

                            style={{

                              borderRadius: '50% 50% 40% 60% / 60% 50% 50% 40%',

                              boxShadow: '4px 6px 0px black',

                              // Delays entrance by 10 frames, adds slight rotation

                              transform: `scale(${calculateSpring(localFrame - 10, 15)}) rotate(-15deg) translateY(${Math.sin(localFrame / 10) * 3}px)`,

                              opacity: interpolate(localFrame - 10, 0, 5, 0, 1)

                            }}

                          />

                        </div>

                      </div>

                    )}


                    {/* 4. HOOKS & OVERLAYS */}

                    {item.trackType === 'overlay' && (

                      item.id.startsWith('s') ? (

                        <div className="absolute top-10 left-10 w-32 h-32">

                          <svg viewBox="0 0 100 100" className="w-full h-full drop-shadow-[0_0_12px_currentColor]" style={{ color: hexColor }}>

                            <circle cx="50" cy="50" r="45" fill="none" stroke="currentColor" strokeWidth="2" strokeDasharray="4 8"

                              style={{ transformOrigin: '50px 50px', transform: `rotate(${localFrame * 2}deg)` }} />

                            <circle cx="50" cy="50" r="35" fill="none" stroke="currentColor" strokeWidth="4" strokeDasharray="220"

                              style={{ strokeDashoffset: interpolate(localFrame, 0, 40, 220, 0) }} />

                          </svg>

                        </div>

                      ) : (

                        <div className="absolute bottom-24 w-full px-6 flex flex-wrap justify-center items-center gap-2">

                          {item.name.split(' ').map((word, i) => {

                            const wordLocalFrame = localFrame - (i * 4);

                            return (

                              <span key={i} className="font-black text-3xl uppercase tracking-tight text-black"

                                style={{

                                  backgroundColor: hexColor, padding: '4px 12px',

                                  transform: `scale(${calculateSpring(wordLocalFrame, 15)}) rotate(${Math.sin(wordLocalFrame / 5) * 3}deg)`,

                                  opacity: interpolate(wordLocalFrame, 0, 5, 0, 1),

                                  boxShadow: `${interpolate(wordLocalFrame, 0, 15, 0, 6)}px ${interpolate(wordLocalFrame, 0, 15, 0, 6)}px 0px white`,

                                  display: 'inline-block'

                                }}>

                                {word}

                              </span>

                            );

                          })}

                        </div>

                      )

                    )}

                  </div>

                );

              })}

              

              {/* Empty State */}

              {visibleItems.length === 0 && (

                <div className="absolute inset-0 flex flex-col items-center justify-center text-gray-600">

                  <Video size={32} className="mb-2 opacity-50" />

                  <span className="text-xs font-medium">Empty Frame</span>

                </div>

              )}

            </div>

          </div>


          {/* AI Prompt Bar */}

          <div className="absolute bottom-6 left-1/2 transform -translate-x-1/2 w-full max-w-xl px-4">

            <form onSubmit={handleAIPrompt} className="relative flex items-center">

              <div className="absolute left-4 text-yellow-500">

                {isGenerating ? <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-yellow-500" /> : <Wand2 size={20} />}

              </div>

              <input 

                type="text" 

                value={prompt}

                onChange={(e) => setPrompt(e.target.value)}

                placeholder="E.g., Generate a stack of organic shapes..."

                className="w-full bg-gray-800/90 backdrop-blur border border-gray-700 text-white rounded-full py-3 pl-12 pr-24 focus:outline-none focus:ring-2 focus:ring-cyan-500 focus:border-transparent shadow-lg"

                disabled={isGenerating}

              />

              <button 

                type="submit"

                disabled={isGenerating || !prompt}

                className="absolute right-2 bg-yellow-500 hover:bg-yellow-400 text-black font-bold py-1.5 px-4 rounded-full text-sm transition disabled:opacity-50"

              >

                Generate

              </button>

            </form>

          </div>

        </main>

      </div>


      {/* --- TIMELINE --- */}

      <div className="h-64 border-t border-gray-800 bg-gray-900 flex flex-col">

        <div className="h-10 border-b border-gray-800 flex items-center px-4 justify-between bg-gray-950">

          <div className="flex items-center gap-4">

            <button onClick={() => setIsPlaying(!isPlaying)} className="text-white hover:text-yellow-500 transition">

              {isPlaying ? <Pause size={18} /> : <Play size={18} />}

            </button>

            <div className="text-xs font-mono text-gray-400">

              FRAME: <span className="text-white">{frame.toString().padStart(3, '0')}</span> / {MAX_FRAMES}

            </div>

          </div>

          <input 

            type="range" min="0" max={MAX_FRAMES} value={frame} 

            onChange={(e) => { setFrame(parseInt(e.target.value)); setIsPlaying(false); }}

            className="w-1/2 accent-yellow-500"

          />

        </div>


        <div className="flex-1 overflow-y-auto overflow-x-hidden relative">

          <div className="absolute top-0 bottom-0 w-0.5 bg-yellow-500 z-50 pointer-events-none" style={{ left: `${(frame / MAX_FRAMES) * 100}%` }}>

            <div className="absolute -top-1 -translate-x-1/2 w-3 h-3 bg-yellow-500 rotate-45" />

          </div>


          {tracks.map((track) => (

            <div key={track.id} className="flex h-16 border-b border-gray-800/50 group">

              <div className="w-48 shrink-0 bg-gray-950 border-r border-gray-800 flex items-center px-3 sticky left-0 z-10">

                <span className="text-xs font-medium text-gray-300 truncate">{track.name}</span>

              </div>

              <div className="flex-1 relative bg-gray-900/50">

                <div className="absolute inset-0 bg-[linear-gradient(to_right,#374151_1px,transparent_1px)] bg-[size:5%_100%] opacity-20" />

                {track.items.map(item => (

                  <div 

                    key={item.id}

                    className={`absolute top-2 bottom-2 ${item.color} rounded text-xs font-bold px-2 flex items-center overflow-hidden cursor-pointer shadow-md border border-white/20 hover:brightness-110 transition`}

                    style={{ left: `${(item.from / MAX_FRAMES) * 100}%`, width: `${(item.duration / MAX_FRAMES) * 100}%` }}

                  >

                    <span className={`truncate ${item.color === 'bg-yellow-500' || item.color === 'bg-yellow-400' ? 'text-black' : 'text-white'}`}>

                      {item.name}

                    </span>

                  </div>

                ))}

              </div>

            </div>

          ))}

        </div>

      </div>

    </div>

  );

}


function TemplateButton({ icon, title, color }) {

  return (

    <button className={`flex items-center gap-3 p-3 bg-gray-950 border border-gray-800 hover:border-gray-600 rounded-lg text-left transition w-full hover:bg-gray-800 border-l-4 ${color}`}>

      <div className="text-gray-400">{icon}</div>

async function fetchWithRetry(url, options, retries = 5) {

  const delays = [1000, 2000, 4000, 8000, 16000];

  for (let i = 0; i < retries; i++) {

    try {

      const response = await fetch(url, options);

      if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

      return await response.json();

    } catch (e) {

      if (i === retries - 1) throw e;

      await new Promise(res => setTimeout(res, delays[i]));

    }

  }

}

async function fetchWithRetry(url, options, retries = 5) {

  const delays = [1000, 2000, 4000, 8000, 16000];

  for (let i = 0; i < retries; i++) {

    try {

      const response = await fetch(url, options);

      if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

      return await response.json();

    } catch (e) {

      if (i === retries - 1) throw e;

      await new Promise(res => setTimeout(res, delays[i]));

    }

  }

}

  const calculateSpring = (localFrame, duration = 15) => {

    if (localFrame < 0) return 0;

    if (localFrame >= duration) return 1;

    const x = localFrame / duration;

    const c1 = 1.70158;

    const c3 = c1 + 1;

    return 1 + c3 * Math.pow(x - 1, 3) + c1 * Math.pow(x - 1, 2);

  };


  // Helper to extract hex colors from our Tailwind strings for custom SVGs/Borders

  const getHexColor = (colorStr) => {

    if (colorStr.includes('yellow')) return '#eab308'; // yellow-500

    if (colorStr.includes('purple')) return '#a855f7'; // purple-500

    if (colorStr.includes('blue')) return '#3b82f6';   // blue-500

    if (colorStr.includes('green')) return '#22c55e';  // green-500

    if (colorStr.includes('red')) return '#ef4444';    // red-500

    return '#ffffff';

  };


  return (

    <div className="flex flex-col h-screen bg-gray-950 text-white font-sans overflow-hidden">

      

      {/* --- TOP BAR --- */}

      <header className="h-14 border-b border-gray-800 flex items-center justify-between px-6 bg-gray-900">

        <div className="flex items-center gap-2">

          <Wand2 className="w-5 h-5 text-yellow-500" />

          <h1 className="font-bold text-lg tracking-tight">CreatorEditor <span className="text-yellow-500">AI</span></h1>

        </div>

        <div className="flex gap-4 text-sm font-medium">

          <button className="px-4 py-1.5 rounded bg-gray-800 hover:bg-gray-700 transition">Export</button>

          <button className="px-4 py-1.5 rounded bg-yellow-500 text-black hover:bg-yellow-400 transition shadow-[2px_2px_0px_white]">Render Video</button>

        </div>

      </header>


      {/* --- MAIN WORKSPACE --- */}

      <div className="flex flex-1 overflow-hidden">

        

        {/* LEFT SIDEBAR: Template Library */}

        <aside className="w-64 border-r border-gray-800 bg-gray-900 flex flex-col hidden md:flex">

          <div className="p-4 border-b border-gray-800 font-bold text-xs uppercase tracking-wider text-gray-400">

            Template Library

          </div>

          <div className="p-4 flex flex-col gap-3 overflow-y-auto">

            <TemplateButton icon={<Type size={16} />} title="Neo-Brutalist Hook" color="border-yellow-500" />

            <TemplateButton icon={<ImageIcon size={16} />} title="3D Parallax PNG" color="border-green-500" />

            <TemplateButton icon={<Layers size={16} />} title="Glassmorphism HUD" color="border-purple-500" />

            <TemplateButton icon={<MousePointer2 size={16} />} title="SVG Target Chase" color="border-blue-500" />

            <TemplateButton icon={<Video size={16} />} title="Cinematic Zoom B-Roll" color="border-red-500" />

          </div>

        </aside>


        {/* CENTER: The Canvas (Remotion Player Placeholder) */}

        <main className="flex-1 flex flex-col bg-black relative">

          

          {/* Video Player Area */}

          <div className="flex-1 flex items-center justify-center p-8">

            {/* 9:16 Aspect Ratio Container */}

            <div className="relative w-full max-w-[320px] aspect-[9/16] bg-gray-900 border border-gray-800 rounded-lg overflow-hidden shadow-2xl ring-1 ring-white/10">

              

              {/* Simulated Render Content based on current frame */}

              {visibleItems.map(item => {

                const localFrame = frame - item.from;

                const hexColor = getHexColor(item.color);


                return (

                  <div key={item.id} className="absolute inset-0 flex items-center justify-center pointer-events-none overflow-hidden">

                    

                    {/* 1. B-ROLL (Animated with Ken Burns slow-zoom) */}

                    {item.trackType === 'video' && (

                      <div 

                        className="absolute inset-0 flex items-center justify-center bg-gray-900 border-4 border-black"

                        style={{ 

                          // Zoom from 1.0x to 1.15x over the duration

                          transform: `scale(${1 + (localFrame * 0.001)})`,

                          opacity: Math.min(localFrame / 10, 1) // 10-frame fade in

                        }}

                      >

                        <div className="absolute inset-0 opacity-20" style={{ backgroundColor: hexColor }} />

                        <span className="opacity-50 text-white font-mono text-xs text-center px-4 drop-shadow-md">

                          B-ROLL LAYER:<br/>{item.name}

                        </span>

                      </div>

                    )}


                    {/* 2. PARALLAX PNG (Animated with Sine-wave floating) */}

                    {item.trackType === 'image' && (

                      <div 

                        className="absolute w-48 h-64 border-2 border-dashed rounded flex items-center justify-center shadow-2xl backdrop-blur-sm"

                        style={{

                          backgroundColor: `${hexColor}33`, // 20% opacity hex

                          borderColor: hexColor,

                          // Float up and down using Sine wave, and slightly scale up over time

                          transform: `translateY(${Math.sin(localFrame / 15) * 15}px) scale(${1 + Math.min(localFrame/200, 0.1)})`,

                          opacity: Math.min(localFrame / 10, 1) // 10-frame fade in

                        }}

                      >

                        <span style={{ color: hexColor }} className="font-bold text-center px-2 drop-shadow-md brightness-150">

                          {item.name}

                        </span>

                      </div>

                    )}


                    {/* 3. HOOKS & OVERLAYS (Spring Math & SVG Rotation) */}

                    {item.trackType === 'overlay' && (

                      item.id.startsWith('s') ? (

                        // SVG HUD TARGET (Math-driven rotation & drawing)

                        <div 

                          className="absolute top-12 left-12 w-24 h-24"

                          style={{

                            // Scale in, then rotate 3 degrees per frame

                            transform: `scale(${calculateSpring(localFrame, 20)}) rotate(${localFrame * 3}deg)`,

                            opacity: Math.min(localFrame / 5, 1)

                          }}

                        >

                          <svg viewBox="0 0 100 100" className="w-full h-full overflow-visible drop-shadow-[0_0_8px_rgba(255,255,255,0.5)]">

                            <circle cx="50" cy="50" r="45" fill="none" stroke={hexColor} strokeWidth="4" strokeDasharray="20 10" />

                            <rect x="48" y="-5" width="4" height="20" fill={hexColor} />

                            <rect x="48" y="85" width="4" height="20" fill={hexColor} />

                            <rect x="-5" y="48" width="20" height="4" fill={hexColor} />

                            <rect x="85" y="48" width="20" height="4" fill={hexColor} />

                          </svg>

                        </div>

                      ) : (

                        // NEO-BRUTALIST TEXT (Spring pop-in + wobble)

                        <div 

                          className="absolute bottom-32 text-black px-4 py-2 font-black text-2xl uppercase text-center max-w-[90%]"

                          style={{

                            backgroundColor: hexColor,

                            boxShadow: '4px 4px 0px #fff',

                            // Calculate bouncy scale-in, then add a continuous slight wobble

                            transform: `scale(${calculateSpring(localFrame, 15)}) rotate(${Math.sin(localFrame / 5) * 2}deg)`,

                            opacity: Math.min(localFrame / 3, 1)

                          }}

                        >

                          {item.name}

                        </div>

                      )

                    )}

                  </div>

                );

              })}

              

              {/* Empty State */}

              {visibleItems.length === 0 && (

                <div className="absolute inset-0 flex flex-col items-center justify-center text-gray-600">

                  <Video size={32} className="mb-2 opacity-50" />

                  <span className="text-xs font-medium">Empty Frame</span>

                </div>

              )}


              {/* HUD / Safe Zones */}

              <div className="absolute inset-0 border-[1px] border-white/5 pointer-events-none m-4 rounded" />

            </div>

          </div>


          {/* AI Prompt Bar (Floating over canvas) */}

          <div className="absolute bottom-6 left-1/2 transform -translate-x-1/2 w-full max-w-xl px-4">

            <form onSubmit={handleAIPrompt} className="relative flex items-center">

              <div className="absolute left-4 text-yellow-500">

                {isGenerating ? <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-yellow-500" /> : <Wand2 size={20} />}

              </div>

              <input 

                type="text" 

                value={prompt}

                onChange={(e) => setPrompt(e.target.value)}

                placeholder="E.g., Add a bouncy text hook right here..."

                className="w-full bg-gray-800/90 backdrop-blur border border-gray-700 text-white rounded-full py-3 pl-12 pr-24 focus:outline-none focus:ring-2 focus:ring-yellow-500 focus:border-transparent shadow-lg"

                disabled={isGenerating}

              />

              <button 

                type="submit"

                disabled={isGenerating || !prompt}

                className="absolute right-2 bg-yellow-500 hover:bg-yellow-400 text-black font-bold py-1.5 px-4 rounded-full text-sm transition disabled:opacity-50"

              >

                Generate

              </button>

            </form>

          </div>

        </main>

      </div>


      {/* --- TIMELINE (The Logic State) --- */}

      <div className="h-64 border-t border-gray-800 bg-gray-900 flex flex-col">

        {/* Timeline Controls */}

        <div className="h-10 border-b border-gray-800 flex items-center px-4 justify-between bg-gray-950">

          <div className="flex items-center gap-4">

            <button onClick={() => setIsPlaying(!isPlaying)} className="text-white hover:text-yellow-500 transition">

              {isPlaying ? <Pause size={18} /> : <Play size={18} />}

            </button>

            <div className="text-xs font-mono text-gray-400">

              FRAME: <span className="text-white">{frame.toString().padStart(3, '0')}</span> / {MAX_FRAMES}

            </div>

          </div>

          <input 

            type="range" 

            min="0" 

            max={MAX_FRAMES} 

            value={frame} 

            onChange={(e) => { setFrame(parseInt(e.target.value)); setIsPlaying(false); }}

            className="w-1/2 accent-yellow-500"

          />

        </div>


        {/* Tracks Area */}

        <div className="flex-1 overflow-y-auto overflow-x-hidden relative">

          

          {/* Playhead Scrubber Line */}

          <div 

            className="absolute top-0 bottom-0 w-0.5 bg-yellow-500 z-50 pointer-events-none"

            style={{ left: `${(frame / MAX_FRAMES) * 100}%` }}

          >

            <div className="absolute -top-1 -translate-x-1/2 w-3 h-3 bg-yellow-500 rotate-45" />

          </div>


          {tracks.map((track, trackIdx) => (

            <div key={track.id} className="flex h-16 border-b border-gray-800/50 group">

              {/* Track Header */}

              <div className="w-48 shrink-0 bg-gray-950 border-r border-gray-800 flex items-center px-3 sticky left-0 z-10">

                <span className="text-xs font-medium text-gray-300 truncate">{track.name}</span>

              </div>

              

              {/* Track Grid */}

              <div className="flex-1 relative bg-gray-900/50">

                {/* Background Grid Lines */}

                <div className="absolute inset-0 bg-[linear-gradient(to_right,#374151_1px,transparent_1px)] bg-[size:5%_100%] opacity-20" />

                

                {/* Clips (Items) */}

                {track.items.map(item => (

                  <div 

                    key={item.id}

                    className={`absolute top-2 bottom-2 ${item.color} rounded text-xs font-bold px-2 flex items-center overflow-hidden cursor-pointer shadow-md border border-white/20 hover:brightness-110 transition`}

                    style={{ 

                      left: `${(item.from / MAX_FRAMES) * 100}%`, 

                      width: `${(item.duration / MAX_FRAMES) * 100}%` 

                    }}

                  >

                    <span className={`truncate ${item.color === 'bg-yellow-500' || item.color === 'bg-yellow-400' ? 'text-black' : 'text-white'}`}>

                      {item.name}

                    </span>

                  </div>

                ))}

              </div>

            </div>

          ))}

        </div>

      </div>

    </div>

  );

}
