@dataclass
class ChartRecord:
    version_folder: str
    file: str
    line: int
    chart_title: str
    chart_type: str
    axis_labels: List[str]
    metric_keys: List[str]
    style_tokens: List[str]
    renderer_family: str
    evidence_snippet: str
    file_mtime: datetime


@dataclass
class FileProfile:
    version_folder: str
    file: str
    records: List[ChartRecord] = field(default_factory=list)
    chart_titles: Set[str] = field(default_factory=set)
    chart_types: Set[str] = field(default_factory=set)
    axis_labels: Set[str] = field(default_factory=set)
    metric_keys: Set[str] = field(default_factory=set)
    style_tokens: Set[str] = field(default_factory=set)
    renderer_family: str = 'Custom'
    file_mtime: datetime = None


def norm(s: str) -> str:
    s = s.lower().strip()
    s = re.sub(r'[^a-z0-9\s\-]+', ' ', s)
    s = re.sub(r'\s+', ' ', s)
    return s


def phrase_overlap_score(want: List[str], corpus: str) -> Tuple[float, List[str], List[str]]:
    c = norm(corpus)
    matched, unmatched = [], []
    for p in [norm(x) for x in want if x.strip()]:
        if p in c:
            matched.append(p)
        else:
            unmatched.append(p)
    score = (len(matched) / len(want)) if want else 0.0
    return score, matched, unmatched


def capture_creation_date(path: Path) -> datetime:
    try:
        out = subprocess.check_output(['mdls', '-raw', '-name', 'kMDItemContentCreationDate', str(path)], text=True).strip()
        if out and out != '(null)':
            return datetime.strptime(out, '%Y-%m-%d %H:%M:%S %z').replace(tzinfo=None)
    except Exception:
        pass
    return datetime.fromtimestamp(path.stat().st_mtime)


def detect_renderer(content: str) -> str:
    lc = content.lower()
    if 'memoizedgooglechart' in lc or 'react-google-charts' in lc or 'googlechart' in lc:
        return 'GoogleChart'
    if 'recharts' in lc or 'responsivecontainer' in lc:
        return 'Recharts'
    if 'chart.js' in lc or 'react-chartjs-2' in lc or 'new chart(' in lc:
        return 'Chart.js'
    return 'Custom'


def detect_type(snippet: str) -> str:
    t = norm(snippet)
    for k, v in TYPE_KEYWORDS.items():
        if k in t:
            return v
    return 'Unknown'


def extract_axis_metrics(snippet: str) -> Tuple[List[str], List[str]]:
    s = norm(snippet)
    axis = sorted({m for m in METRIC_TERMS if m in s})
    return axis, axis.copy()


def is_chart_ui_file(path: Path, content: str) -> bool:
    pl = str(path).lower()
    if any(h in pl for h in UI_FILE_HINTS):
        return True
    c = norm(content)
    return sum(1 for t in KNOWN_CHART_LABELS if t in c) >= 2


def scan_root(root: Path, allow_fallback: bool = False) -> List[ChartRecord]:
    out: List[ChartRecord] = []
    for dirpath, dirnames, filenames in os.walk(root):
        dirnames[:] = [d for d in dirnames if d not in SKIP_DIRS]
        for fn in filenames:
            p = Path(dirpath) / fn
            if p.suffix.lower() not in EXTS:
                continue
            try:
                txt = p.read_text(errors='ignore')
            except Exception:
                continue

            if not is_chart_ui_file(p, txt):
                continue
            if not allow_fallback:
                if '/views/' not in str(p).replace('\\', '/') and '/components/' not in str(p).replace('\\', '/'):
                    continue

            lines = txt.splitlines()
            renderer = detect_renderer(txt)
            mtime = datetime.fromtimestamp(p.stat().st_mtime)
            version = p.relative_to(root).parts[0] if root == APP_COPIES_ROOT else p.parent.name

            title_hits = 0
            for i, line in enumerate(lines, start=1):
                if not re.search(r'(title\s*[:=]|chart|graph|retention|ctr|impressions|watch time|subscribers)', line, re.I):
                    continue
                s = max(1, i - 2)
                e = min(len(lines), i + 2)
                snippet = '\n'.join(lines[s - 1:e])

                title = ''
                for pat in TITLE_PATTERNS:
                    m = pat.search(line) or pat.search(snippet)
                    if m:
                        title = m.group(1).strip()
                        title_hits += 1
                        break

                ctype = detect_type(snippet)
                axis, metrics = extract_axis_metrics(snippet)
                styles = sorted(set(STYLE_RE.findall(snippet)))

                if not title and ctype == 'Unknown' and not axis:
                    continue

                out.append(ChartRecord(
                    version_folder=version,
                    file=str(p),
                    line=i,
                    chart_title=title,
                    chart_type=ctype,
                    axis_labels=axis,
                    metric_keys=metrics,
                    style_tokens=styles,
                    renderer_family=renderer,
                    evidence_snippet=' '.join(snippet.split())[:240],
                    file_mtime=mtime,
                ))

            if title_hits == 0 and not allow_fallback:
                # keep only files where chart titles are explicit; drop diffuse matches
                out = [r for r in out if r.file != str(p)]
    return out


def build_profiles(records: List[ChartRecord]) -> Dict[str, FileProfile]:
    d: Dict[str, FileProfile] = {}
    for r in records:
        if r.file not in d:
            d[r.file] = FileProfile(r.version_folder, r.file, renderer_family=r.renderer_family, file_mtime=r.file_mtime)
        p = d[r.file]
        p.records.append(r)
        if r.chart_title:
            p.chart_titles.add(r.chart_title)
        if r.chart_type != 'Unknown':
            p.chart_types.add(r.chart_type)
        p.axis_labels.update(r.axis_labels)
        p.metric_keys.update(r.metric_keys)
        p.style_tokens.update(r.style_tokens)
        if p.renderer_family == 'Custom' and r.renderer_family != 'Custom':
            p.renderer_family = r.renderer_family
    return d

def best_line(profile: FileProfile, labels: List[str]) -> int:
    want = [norm(x) for x in labels]
    top = (0.0, 1)
    for r in profile.records:
        corpus = f"{r.chart_title} {r.evidence_snippet}"
        s, _, _ = phrase_overlap_score(want, corpus)
        if s > top[0]:
            top = (s, r.line)
    return top[1]


def score(profile: FileProfile, shot_dt: datetime, clues: Dict[str, List[str]]) -> Tuple[float, Dict[str, float], List[str], List[str], int]:
    title_corpus = ' | '.join(sorted(profile.chart_titles)) + ' | ' + profile.file + ' | ' + ' '.join(r.evidence_snippet for r in profile.records[:30])
    title_score, matched, unmatched = phrase_overlap_score(clues['labels'], title_corpus)

    type_score, _, _ = phrase_overlap_score(clues['types'], ' '.join(sorted(profile.chart_types)))
    metric_score, _, _ = phrase_overlap_score(clues['metrics'], ' '.join(sorted(profile.metric_keys | profile.axis_labels)))

    days = abs((shot_dt - profile.file_mtime).total_seconds()) / 86400.0
    date_score = max(0.0, 1.0 - (days / 10.0))

    final = (
        WEIGHTS['title'] * title_score +
        WEIGHTS['type'] * type_score +
        WEIGHTS['metric'] * metric_score +
        WEIGHTS['date'] * date_score
    )

    # Strong anchor preference for expected files.
    if profile.file in HIGH_SIGNAL_ANCHORS:
        final = min(1.0, final + 0.08)

    line = best_line(profile, clues['labels'])
    br = {
        'title_label': round(title_score, 4),
        'type_layout': round(type_score, 4),
        'metric_axis': round(metric_score, 4),
        'date_proximity': round(date_score, 4),
        'weighted_final': round(final, 4),
    }
    return final, br, matched, unmatched, line


def relative_version(file_path: str) -> str:
    p = Path(file_path)
    try:
        return p.relative_to(APP_COPIES_ROOT).parts[0]
    except Exception:
        return p.parent.name


def scan_downloads_fallback() -> Dict[str, FileProfile]:
    records = scan_root(DOWNLOADS_ROOT, allow_fallback=True)
    profiles = build_profiles(records)
    profiles = {
        k: v for k, v in profiles.items()
        if '/APP_COPIES/' not in k
        and ('researchlab' in v.file.lower() or 'datavizualizations' in v.file.lower() or 'mastergraphspage' in v.file.lower() or 'chartengine' in v.file.lower())
        and (len(v.chart_titles & KNOWN_CHART_LABELS) >= 1 or len(v.chart_titles) >= 3)
    }
    return profiles
