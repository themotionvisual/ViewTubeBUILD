const VelocityEqualizer = () => {
  const [eq, setEq] = useState([40, 70, 45, 90, 60, 80, 50, 100, 30]);
  
  const randomize = () => {
    setEq(eq.map(() => Math.floor(Math.random() * 80) + 20));
  };

  return (
    <div className="flex-1 border-[4px] border-black bg-white p-4 rounded-xl flex items-end gap-2 relative cursor-pointer group" onClick={randomize}>
      <div className="absolute top-3 left-4 font-black uppercase text-[12px] bg-[#FFE357] px-3 py-1 border-[3px] border-black rounded-lg shadow-[2px_2px_0px_0px_black] z-10">Velocity</div>
      {eq.map((val, i) => (
        <div key={i} className="flex-1 bg-[#24D3FF] border-[3px] border-black rounded-t-md group-hover:bg-[#FF7497] transition-all duration-300" style={{ height: `${val}%` }} />
      ))}
    </div>
  );
};

const ResonanceGrid = () => {
  const [grid, setGrid] = useState(Array(16).fill(false));
  const toggle = (i) => {
    const newGrid = [...grid];
    newGrid[i] = !newGrid[i];
    setGrid(newGrid);
  };
  
  return (
    <div className="flex-1 border-[4px] border-black bg-white p-4 rounded-xl shadow-[4px_4px_0px_0px_black] flex flex-col">
      <div className="font-black uppercase text-[10px] mb-3 border-b-[3px] border-black pb-1">Resonance</div>
      <div className="flex-1 grid grid-cols-4 gap-2">
        {grid.map((active, i) => (
          <div 
            key={i} 
            onClick={() => toggle(i)}
            className={`border-[2px] border-black rounded-sm cursor-pointer transition-colors ${active ? 'bg-[#FF7497]' : 'bg-[#A0E8AF]'}`} 
          />
        ))}
      </div>
    </div>
  );
};

const SequenceEngine = () => {
  const [power, setPower] = useState(false);
  const [freq, setFreq] = useState(0);
  const [launched, setLaunched] = useState(false);

  const handleLaunch = () => {
    if (power && freq >= 95) setLaunched(true);
  };

  const reset = () => {
    setPower(false); setFreq(0); setLaunched(false);
  };

  return (
    <div className={`p-6 border-[4px] border-black rounded-xl flex flex-col gap-6 shadow-[8px_8px_0px_0px_black] transition-colors duration-1000 ${launched ? 'bg-[#CCFF00]' : 'bg-white'}`}>
      <div className="flex justify-between items-center border-b-[4px] border-black pb-4">
        <h3 className="font-[1000] uppercase text-2xl tracking-tighter">{launched ? 'SEQUENCE ACTIVE' : 'CHAIN REACTION'}</h3>
        {launched && <button onClick={reset} className="font-black text-[10px] bg-black text-white px-3 py-1 rounded-full uppercase">Reset</button>}
      </div>
      
      <div className="flex items-center gap-6">
        {/* STEP 1: Power */}
        <div className="flex flex-col items-center gap-2">
          <div className="text-[10px] font-black uppercase">1. Power</div>
          <button 
            onClick={() => { setPower(!power); if (power) setFreq(0); }} 
            className={`w-16 h-24 border-[4px] border-black rounded-xl flex flex-col p-1 transition-colors ${power ? 'bg-black' : 'bg-gray-200'}`}
          >
            <div className={`w-full h-1/2 border-[3px] border-black rounded-lg transition-transform duration-300 ${power ? 'bg-[#FF3399] translate-y-full' : 'bg-white'}`} />
          </button>
        </div>

        <FastForward size={32} className={power ? "text-black" : "text-black/20"} />

        {/* STEP 2: Frequency Slider */}
        <div className="flex-1 flex flex-col gap-2">
          <div className="flex justify-between text-[10px] font-black uppercase">
            <span>2. Charge Drive</span>
            <span className={power ? "text-[#24D3FF]" : "text-black/30"}>{freq}%</span>
          </div>
          <div className={`h-[48px] border-[4px] border-black rounded-xl px-4 flex flex-col justify-center transition-colors ${power ? 'bg-[#FFE357]' : 'bg-gray-200 pointer-events-none opacity-50'}`}>
            <MechanicalSlider value={freq} onChange={setFreq} activeColor="#24D3FF" />
          </div>
        </div>

        <FastForward size={32} className={freq >= 95 ? "text-black" : "text-black/20"} />

        {/* STEP 3: Launch */}
        <div className="flex flex-col items-center gap-2">
          <div className="text-[10px] font-black uppercase">3. Engage</div>
          <button 
            onClick={handleLaunch}
            disabled={!power || freq < 95 || launched}
            className={`w-24 h-24 border-[4px] border-black rounded-full font-[1000] text-xl transition-all shadow-[4px_4px_0px_0px_black]
              ${power && freq >= 95 && !launched ? 'bg-[#FF3399] text-white hover:scale-105 active:translate-y-1 active:shadow-none cursor-pointer animate-pulse' : 'bg-gray-100 text-black/30 cursor-not-allowed'}
              ${launched ? 'bg-black text-[#CCFF00] shadow-none scale-95' : ''}
            `}
          >
            {launched ? 'ON' : 'GO'}
          </button>
        </div>
      </div>
    </div>
  );
};
