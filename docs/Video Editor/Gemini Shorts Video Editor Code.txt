const callGemini = async (prompt: string, systemInstruction: string, schema?: any) => {
  const apiKey = "";
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
  const payload: any = { contents: [{ parts: [{ text: prompt }] }], systemInstruction: { parts: [{ text: systemInstruction }] } };
  if (schema) payload.generationConfig = { responseMimeType: "application/json", responseSchema: schema };

  for (let i = 0; i < 3; i++) {
    try {
      const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
      if (!response.ok) throw new Error(`API Error: ${response.status}`);
      const data = await response.json();
      const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
      return schema ? JSON.parse(text) : text;
    } catch (e) {
      if (i === 2) throw e;
      await new Promise(res => setTimeout(res, 2000));
    }
  }
};

const callImagen = async (prompt: string) => {
  const apiKey = "";
  const url = `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`;
  try {
    const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ instances: { prompt }, parameters: { sampleCount: 1 } }) });
    const data = await response.json();
    if (data.predictions?.[0]) return `data:image/png;base64,${data.predictions[0].bytesBase64Encoded}`;
    throw new Error("No image");
  } catch (e) {
    return `https://images.unsplash.com/photo-1618005182384-a83a8bd57fbe?q=80&w=1000&auto=format&fit=crop`; 
  }
};

const pcmToWav = (pcmData: string, sampleRate: number) => {
  const binaryString = atob(pcmData);
  const bytes = new Uint8Array(binaryString.length);
  for (let i = 0; i < binaryString.length; i++) bytes[i] = binaryString.charCodeAt(i);
  const wavBuffer = new ArrayBuffer(44 + bytes.length);
  const view = new DataView(wavBuffer);
  const writeString = (v: DataView, o: number, s: string) => { for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); };
  writeString(view, 0, 'RIFF'); view.setUint32(4, 36 + bytes.length, true); writeString(view, 8, 'WAVE');
  writeString(view, 12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);
  view.setUint16(22, 1, true); view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);
  view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);
  new Uint8Array(wavBuffer, 44).set(bytes);
  return URL.createObjectURL(new Blob([wavBuffer], { type: 'audio/wav' }));
};

const callGeminiTTS = async (textToSpeak: string) => {
  const apiKey = "";
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${apiKey}`;
  const payload = { contents: [{ parts: [{ text: textToSpeak }] }], generationConfig: { responseModalities: ["AUDIO"], speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Zephyr" } } } } };
  try {
    const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
    const data = await response.json();
    const inlineData = data.candidates?.[0]?.content?.parts?.[0]?.inlineData;
    if (!inlineData) return null;
    const match = inlineData.mimeType.match(/rate=(\d+)/);
    return pcmToWav(inlineData.data, match ? parseInt(match[1]) : 24000);
  } catch (e) { return null; }
};


// --- TYPES & DETERMINISTIC ENGINE ---
type LayerType = 'media' | 'text' | 'audio';

interface Keyframe {
  time: number;
  props: { x: number; y: number; scale: number; rotation: number; opacity: number; blur: number; hsb: { hue: number; sat: number; bri: number } };
}

interface Layer {
  id: string; name: string; url: string; type: LayerType; isVisible: boolean; color?: string;
  start: number; end: number;
  x: number; y: number; scale: number; rotation: number; opacity: number; blur: number; hsb: { hue: number; sat: number; bri: number };
  entryAnim: 'none' | 'fade' | 'pop' | 'slideUp' | 'glitch';
  motionAnim: 'none' | 'panRight' | 'zoomIn' | 'float';
  zIndex: number; keyframes: Keyframe[];
}

const lerp = (start: number, end: number, t: number) => start + (end - start) * t;

// Gets the exact mathematical properties of a layer at a given time
const getInterpolatedProps = (layer: Layer, time: number) => {
  const baseProps = { x: layer.x, y: layer.y, scale: layer.scale, rotation: layer.rotation, opacity: layer.opacity, blur: layer.blur, hsb: { ...layer.hsb } };
  if (!layer.keyframes || layer.keyframes.length === 0) return baseProps;

  const kfs = [...layer.keyframes].sort((a, b) => a.time - b.time);
  const prev = kfs.slice().reverse().find(k => k.time <= time);
  const next = kfs.find(k => k.time > time);

  if (prev && !next) return JSON.parse(JSON.stringify(prev.props));
  if (!prev && next) return baseProps;
  if (prev && next) {
    const t = (time - prev.time) / (next.time - prev.time);
    return {
      x: lerp(prev.props.x, next.props.x, t), y: lerp(prev.props.y, next.props.y, t),
      scale: lerp(prev.props.scale, next.props.scale, t), rotation: lerp(prev.props.rotation, next.props.rotation, t),
      opacity: lerp(prev.props.opacity, next.props.opacity, t), blur: lerp(prev.props.blur, next.props.blur, t),
      hsb: { hue: lerp(prev.props.hsb.hue, next.props.hsb.hue, t), sat: lerp(prev.props.hsb.sat, next.props.hsb.sat, t), bri: lerp(prev.props.hsb.bri, next.props.hsb.bri, t) }
    };
  }
  return baseProps;
};

// Returns the final styling for the canvas based on interpolation + entry/motion animations
const getLayerStyles = (layer: Layer, time: number, maxTime: number) => {
  const interp = getInterpolatedProps(layer, time);
  
  let props = {
    x: interp.x, y: interp.y, scale: interp.scale, rotate: interp.rotation, opacity: interp.opacity,
    filter: `hue-rotate(${interp.hsb.hue}deg) saturate(${interp.hsb.sat}%) brightness(${interp.hsb.bri}%) blur(${interp.blur}px)`
  };

  const timeInClip = time - layer.start;
  const dur = layer.end - layer.start;
  if (timeInClip < 0) return props; // Handled by AnimatePresence mounting

  // Entry Animations
  const entryDur = Math.min(0.5, dur);
  const ep = Math.min(1, Math.max(0, timeInClip / entryDur));
  const e = 1 - Math.pow(1 - ep, 3); // easeOutCubic

  if (layer.entryAnim === 'fade') props.opacity *= e;
  if (layer.entryAnim === 'pop') { props.scale *= (0.5 + 0.5 * e); props.opacity *= e; }
  if (layer.entryAnim === 'slideUp') { props.y += 100 * (1 - e); props.opacity *= e; }
  if (layer.entryAnim === 'glitch' && e < 1) { props.x += (Math.random() * 20 - 10) * (1 - e); props.opacity *= e; }

  // Continuous Motion
  if (layer.motionAnim !== 'none' && (!layer.keyframes || layer.keyframes.length === 0)) {
    const mp = timeInClip / maxTime;
    if (layer.motionAnim === 'panRight') props.x += 100 * mp;
    if (layer.motionAnim === 'zoomIn') props.scale *= (1 + 0.2 * mp);
    if (layer.motionAnim === 'float') props.y += Math.sin(timeInClip * 2) * 20;
  }

  return props;
};

export const ShortsStudio = () => {
  const { addProject } = useBrain();
  
  // Layout State
  const [aspectRatio, setAspectRatio] = useState<'9:16' | '16:9'>('9:16');
  const [activeTool, setActiveTool] = useState<'ai' | 'media' | 'captions' | 'audio' | 'fx'>('media');
  
  // Editor State
  const [maxTime, setMaxTime] = useState(15);
  const [currentTime, setCurrentTime] = useState(0); 
  const [isPlaying, setIsPlaying] = useState(false);
  const [isSnapping, setIsSnapping] = useState(true);
  const [timelineZoom, setTimelineZoom] = useState(1);
  const [activeLayerId, setActiveLayerId] = useState<string | null>('bg-1');
  
  const [aiPrompt, setAiPrompt] = useState('');
  const [captionPrompt, setCaptionPrompt] = useState('');
  const [fxPrompt, setFxPrompt] = useState('');
  const [isGenerating, setIsGenerating] = useState(false);
  const [toastMsg, setToastMsg] = useState<string | null>(null);

  const [layers, setLayers] = useState<Layer[]>([
    { 
      id: 'bg-1', name: 'Background', 
      url: 'https://images.unsplash.com/photo-1618005182384-a83a8bd57fbe?q=80&w=1000&auto=format&fit=crop', 
      type: 'media', isVisible: true, start: 0, end: 15,
      x: 0, y: 0, scale: 1.1, rotation: 0, opacity: 1, blur: 0, zIndex: 1,
      hsb: { hue: 0, sat: 100, bri: 100 },
      entryAnim: 'fade', motionAnim: 'zoomIn', keyframes: []
    }
  ]);

  const trackAreaRef = useRef<HTMLDivElement>(null);

  // --- PLAYBACK ENGINE ---
  useEffect(() => {
    let req: number;
    let lastTime = performance.now();
    const loop = (now: number) => {
      if (isPlaying) {
        const dt = (now - lastTime) / 1000;
        setCurrentTime(prev => {
          let next = prev + dt;
          if (next >= maxTime) { setIsPlaying(false); return 0; }
          return next;
        });
      }
      lastTime = now;
      req = requestAnimationFrame(loop);
    };
    req = requestAnimationFrame(loop);
    return () => cancelAnimationFrame(req);
  }, [isPlaying, maxTime]);

  // --- AUDIO ENGINE ---
  useEffect(() => {
    layers.filter(l => l.type === 'audio').forEach(layer => {
      const audioEl = document.getElementById(`audio-${layer.id}`) as HTMLAudioElement;
      if (audioEl) {
        if (isPlaying && currentTime >= layer.start && currentTime <= layer.end) {
          if (audioEl.paused) {
            const targetTime = currentTime - layer.start;
            if (Math.abs(audioEl.currentTime - targetTime) > 0.2) audioEl.currentTime = targetTime;
            audioEl.play().catch(() => {});
          }
        } else if (!audioEl.paused) {
          audioEl.pause();
        }
      }
    });
  }, [isPlaying, currentTime, layers]);

  const showToast = (msg: string) => { setToastMsg(msg); setTimeout(() => setToastMsg(null), 3000); };

  // --- MAGNETIC SNAPPING LOGIC ---
  const getSnappedTime = (time: number, ignoreLayerId: string | null = null) => {
    if (!isSnapping) return time;
    const threshold = 0.4;
    const snapPoints = [
      ...Array.from({length: Math.ceil(maxTime) + 1}, (_, i) => i), // Seconds
      ...layers.filter(l => l.id !== ignoreLayerId).flatMap(l => [l.start, l.end]), // Layer edges
      0, maxTime
    ];
    const closest = snapPoints.reduce((prev, curr) => Math.abs(curr - time) < Math.abs(prev - time) ? curr : prev, time);
    return Math.abs(closest - time) < threshold ? closest : time;
  };

  // --- LAYER MANAGEMENT ---
  const addLayer = (name: string, url: string, type: LayerType, color?: string, start = 0, end = maxTime, motionAnim: any = 'none') => {
    const topZ = layers.length > 0 ? Math.max(...layers.map(l => l.zIndex)) : 0;
    const newLayer: Layer = {
      id: Math.random().toString(36).substr(2, 9), name, url, type, color, start, end, isVisible: true,
      x: 0, y: 0, scale: 1, rotation: 0, opacity: 1, blur: 0, zIndex: topZ + 1,
      hsb: { hue: 0, sat: 100, bri: 100 },
      entryAnim: type === 'text' ? 'pop' : 'fade', motionAnim, keyframes: []
    };
    setLayers(prev => [...prev, newLayer]);
    setActiveLayerId(newLayer.id);
  };

  const updateLayer = (id: string, updates: Partial<Layer>) => {
    setLayers(layers.map(l => l.id === id ? { ...l, ...updates } : l));
  };

  const deleteLayer = (id: string) => {
    setLayers(layers.filter(l => l.id !== id));
    if (activeLayerId === id) setActiveLayerId(null);
  };

  const handleCut = () => {
    if (!activeLayerId) { showToast("Select a layer to cut."); return; }
    const layer = layers.find(l => l.id === activeLayerId);
    if (!layer) return;
    if (currentTime <= layer.start || currentTime >= layer.end) {
      showToast("Playhead must be over the selected clip to cut."); return;
    }
    
    const secondHalf: Layer = {
      ...JSON.parse(JSON.stringify(layer)), // Deep copy
      id: Math.random().toString(36).substr(2, 9),
      start: currentTime,
    };
    
    setLayers(prev => {
      const updatedPrev = prev.map(l => l.id === layer.id ? { ...l, end: currentTime } : l);
      return [...updatedPrev, secondHalf];
    });
    showToast("✂️ Clip Split!");
  };

  // --- LAYER DRAG & STRETCH LOGIC ---
  const [dragAction, setDragAction] = useState<{ id: string, type: 'start' | 'end' | 'move', initialStart: number, initialEnd: number, startX: number } | null>(null);

  const handlePointerDown = (e: any, id: string, type: 'start' | 'end' | 'move') => {
    e.stopPropagation();
    const layer = layers.find(l => l.id === id);
    if (!layer) return;
    setDragAction({ id, type, initialStart: layer.start, initialEnd: layer.end, startX: e.clientX });
    setActiveLayerId(id);
  };

  useEffect(() => {
    const handleMove = (e: PointerEvent) => {
      if (!dragAction || !trackAreaRef.current) return;
      const rect = trackAreaRef.current.getBoundingClientRect();
      const deltaX = e.clientX - dragAction.startX;
      const timeShift = (deltaX / rect.width) * maxTime;

      const layer = layers.find(l => l.id === dragAction.id);
      if (!layer) return;

      if (dragAction.type === 'move') {
          let newStart = dragAction.initialStart + timeShift;
          let newEnd = dragAction.initialEnd + timeShift;
          if (isSnapping) {
             const snapped = getSnappedTime(newStart, layer.id);
             const snapDiff = snapped - newStart;
             newStart += snapDiff;
             newEnd += snapDiff;
          }
          if (newStart < 0) { newEnd -= newStart; newStart = 0; }
          if (newEnd > maxTime) { newStart -= (newEnd - maxTime); newEnd = maxTime; }
          updateLayer(layer.id, { start: newStart, end: newEnd });
      } else if (dragAction.type === 'start') {
          let newStart = dragAction.initialStart + timeShift;
          if (isSnapping) newStart = getSnappedTime(newStart, layer.id);
          newStart = Math.min(newStart, layer.end - 0.1);
          newStart = Math.max(0, newStart);
          updateLayer(layer.id, { start: newStart });
      } else if (dragAction.type === 'end') {
          let newEnd = dragAction.initialEnd + timeShift;
          if (isSnapping) newEnd = getSnappedTime(newEnd, layer.id);
          newEnd = Math.max(newEnd, layer.start + 0.1);
          newEnd = Math.min(maxTime, newEnd);
          updateLayer(layer.id, { end: newEnd });
      }
    };
    const handleUp = () => setDragAction(null);

    if (dragAction) {
       window.addEventListener('pointermove', handleMove);
       window.addEventListener('pointerup', handleUp);
    }
    return () => {
       window.removeEventListener('pointermove', handleMove);
       window.removeEventListener('pointerup', handleUp);
    };
  }, [dragAction, isSnapping, layers, maxTime]);

  // --- AUTO-KEYFRAME LOGIC ---
  const handlePropChange = (propPath: string, value: number) => {
    const layer = layers.find(l => l.id === activeLayerId);
    if (!layer) return;

    const hasKfs = layer.keyframes && layer.keyframes.length > 0;
    if (hasKfs) {
      const kfs = [...layer.keyframes];
      const existingIdx = kfs.findIndex(k => Math.abs(k.time - currentTime) < 0.05);
      
      if (existingIdx >= 0) {
        if (propPath.includes('.')) {
          const [obj, key] = propPath.split('.');
          (kfs[existingIdx].props as any)[obj][key] = value;
        } else (kfs[existingIdx].props as any)[propPath] = value;
      } else {
        const interp = JSON.parse(JSON.stringify(getInterpolatedProps(layer, currentTime)));
        if (propPath.includes('.')) {
          const [obj, key] = propPath.split('.');
          interp[obj][key] = value;
        } else interp[propPath] = value;
        kfs.push({ time: currentTime, props: interp });
      }
      updateLayer(layer.id, { keyframes: kfs, motionAnim: 'none' }); 
    } else {
      if (propPath.includes('.')) {
        const [obj, key] = propPath.split('.');
        updateLayer(layer.id, { [obj]: { ...(layer as any)[obj], [key]: value } });
      } else updateLayer(layer.id, { [propPath]: value });
    }
  };

  const toggleKeyframe = (layerId: string) => {
    const layer = layers.find(l => l.id === layerId);
    if (!layer) return;
    const kfs = [...(layer.keyframes || [])];
    const existingIdx = kfs.findIndex(k => Math.abs(k.time - currentTime) < 0.05);
    if (existingIdx >= 0) kfs.splice(existingIdx, 1);
    else kfs.push({ time: currentTime, props: getInterpolatedProps(layer, currentTime) });
    updateLayer(layerId, { keyframes: kfs, motionAnim: 'none' });
  };

    if (!aiPrompt) return;

        type: "OBJECT", properties: {

          }, required: ["name", "type", "content"] }}

      if (res && res.layers) {

        }
        setAiPrompt(''); showToast("Assets deployed.");
      }
    } catch(e) { showToast("Generation failed."); } finally { setIsGenerating(false); }
  };

  const handleGenerateTTS = async () => {

    setIsGenerating(true); showToast("Synthesizing Voice...");

      const audioUrl = await callGeminiTTS(aiPrompt);

    if (!aiPrompt) return;
    setIsGenerating(true); showToast("Writing Hook...");

  };

    if (!captionPrompt) return;
    setIsGenerating(true); showToast("Writing Captions...");
    try {
      const schema = { type: "OBJECT", properties: { captions: { type: "ARRAY", items: { type: "OBJECT", properties: { text: { type: "STRING" }, start: { type: "NUMBER" }, end: { type: "NUMBER" }, color: { type: "STRING" } }, required: ["text", "start", "end", "color"] }} } };
      const res = await callGemini(captionPrompt, "Write 1-3 word captions sequentially across 15 seconds based on the topic.", schema);
      if (res && res.captions) {
        res.captions.forEach((cap: any) => addLayer(`Cap: ${cap.text}`, cap.text, 'text', cap.color || '#FFFFFF', cap.start, cap.end, 'none'));
        setCaptionPrompt(''); showToast("Captions added.");
      }
    } catch(e) { showToast("Captions failed."); } finally { setIsGenerating(false); }
  };

  const handleSmartGrade = async () => {
    if (!fxPrompt || !activeLayerId) return;
    setIsGenerating(true); showToast("Analyzing Grade...");
    try {
      const schema = { type: "OBJECT", properties: { hue: { type: "NUMBER" }, sat: { type: "NUMBER" }, bri: { type: "NUMBER" }, blur: { type: "NUMBER" } }, required: ["hue", "sat", "bri", "blur"] };
      const res = await callGemini(fxPrompt, "Translate mood to HSB values (Hue 0-360, Sat 0-200, Bri 0-200, Blur 0-20).", schema);
      if (res) { updateLayer(activeLayerId, { hsb: { hue: res.hue, sat: res.sat, bri: res.bri }, blur: res.blur }); setFxPrompt(''); showToast(`Grade Applied`); }
    } catch(e) { showToast("Grade failed."); } finally { setIsGenerating(false); }
  };

  const activeLayer = layers.find(l => l.id === activeLayerId);
  const activeInterpProps = activeLayer ? getInterpolatedProps(activeLayer, currentTime) : null;
  const hasKeyframeAtCurrent = activeLayer?.keyframes?.some(k => Math.abs(k.time - currentTime) < 0.05);

  // --- SUB-COMPONENTS ---
  const VerticalNavMenu = () => (
    <div className="w-24 bg-black flex flex-col items-center py-6 gap-4 z-20 shrink-0 shadow-[8px_0_20px_rgba(0,0,0,0.3)]">
      {[
        { id: 'media', icon: ImageIcon, label: 'Media' },
        { id: 'ai', icon: Wand, label: 'AI Gen' },

            activeTool === tool.id ? 'bg-[#CCFF00] text-black border-[3px] border-black shadow-[4px_4px_0px_0px_#FF3399]' : 'bg-white/5 text-white hover:bg-white/10'
          }`}

  const ToolPanel = () => (
    <div className="w-80 bg-white border-r-[4px] border-black flex flex-col z-10 shrink-0 shadow-[4px_0px_0px_0px_rgba(0,0,0,1)]">
      <div className="flex-1 p-6 overflow-y-auto">
        {activeTool === 'ai' && (
          <div className="space-y-4">
            <h3 className="font-black uppercase text-xl border-b-[3px] border-black pb-2 flex items-center gap-2"><Wand/> AI Director</h3>
            <p className="text-xs font-bold text-gray-500">Describe a cinematic scene. The AI generates high-res visuals and motion.</p>
            <textarea value={aiPrompt} onChange={e => setAiPrompt(e.target.value)} disabled={isGenerating} className="w-full h-32 p-3 border-[3px] border-black font-bold resize-none shadow-[4px_4px_0px_0px_black] focus:outline-none focus:translate-y-1 focus:shadow-none transition-all" placeholder="e.g. A floating crystal skull..." />
            <div className="flex gap-2">
              <button onClick={handleAiGenerate} disabled={isGenerating} className="flex-[2] py-3 bg-black text-[#00CCFF] border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_#00CCFF] active:translate-y-1 active:shadow-none transition-all flex justify-center items-center gap-2">
                <Star size={16}/> {isGenerating ? 'Rendering...' : 'Generate'}
              </button>
              <button onClick={handleMagicHook} disabled={isGenerating} className="flex-1 py-3 bg-[#FFDD00] text-black border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_black] active:translate-y-1 active:shadow-none transition-all flex justify-center items-center text-[10px]">
                ✨ Hook
              </button>
            </div>
          </div>
        )}
        
        {activeTool === 'captions' && (
          <div className="space-y-4">
            <h3 className="font-black uppercase text-xl border-b-[3px] border-black pb-2 flex items-center gap-2"><Type/> AI Captions</h3>
            <textarea value={captionPrompt} onChange={e => setCaptionPrompt(e.target.value)} disabled={isGenerating} className="w-full h-24 p-3 border-[3px] border-black font-bold resize-none shadow-[4px_4px_0px_0px_black] focus:outline-none" placeholder="e.g. 3 facts about space..." />
            <button onClick={handleAutoCaptions} disabled={isGenerating} className="w-full py-3 bg-[#CCFF00] text-black border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_black] active:translate-y-1 active:shadow-none transition-all flex justify-center gap-2">
              <Star size={18}/> {isGenerating ? 'Writing...' : 'Generate Captions'}
            </button>
          </div>
        )}

        {activeTool === 'fx' && (
          <div className="space-y-4">
            <h3 className="font-black uppercase text-xl border-b-[3px] border-black pb-2 flex items-center gap-2"><Settings/> Smart Grade</h3>
            <p className="text-[10px] font-bold text-gray-500">Select a layer, type a mood, let AI grade it.</p>
            <input type="text" value={fxPrompt} onChange={e => setFxPrompt(e.target.value)} disabled={isGenerating} className="w-full p-3 border-[3px] border-black font-bold shadow-[4px_4px_0px_0px_black] focus:outline-none" placeholder="e.g. Matrix Green..." />
            <button onClick={handleSmartGrade} disabled={isGenerating || !activeLayer} className="w-full py-3 bg-black text-[#FF3399] border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_#FF3399] active:translate-y-1 active:shadow-none transition-all disabled:opacity-50">
              Apply Grade
            </button>
          </div>
        )}

        {activeTool === 'media' && (
          <div className="space-y-4">
             <h3 className="font-black uppercase text-xl border-b-[3px] border-black pb-2 flex items-center gap-2"><ImageIcon/> Stock Pool</h3>
             <button className="w-full py-3 bg-[#CCFF00] border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_black] active:translate-y-1 active:shadow-none flex justify-center gap-2"><UploadCloud size={18} /> Upload Media</button>
             <div className="grid grid-cols-2 gap-3 mt-4">
                {["https://images.unsplash.com/photo-1514565131-fce0801e5785?q=80&w=800", "https://images.unsplash.com/photo-1550745165-9bc0b252726f?q=80&w=800"].map((url, i) => (
                  <img key={i} src={url} onClick={() => addLayer(`Stock ${i}`, url, 'media')} className="w-full aspect-square object-cover border-[3px] border-black cursor-pointer hover:scale-105 hover:shadow-[4px_4px_0px_0px_#FF3399] transition-all" alt="stock" />
                ))}
             </div>
          </div>
        )}

        {activeTool === 'audio' && (
          <div className="space-y-4">
            <h3 className="font-black uppercase text-xl border-b-[3px] border-black pb-2 flex items-center gap-2"><Mic/> TTS Voice</h3>
            <textarea value={aiPrompt} onChange={e => setAiPrompt(e.target.value)} disabled={isGenerating} className="w-full h-32 p-3 border-[3px] border-black font-bold resize-none shadow-[4px_4px_0px_0px_black] focus:outline-none" placeholder="Type script..." />
            <button onClick={handleGenerateTTS} disabled={isGenerating} className="w-full py-3 bg-[#FF3399] text-white border-[3px] border-black font-black uppercase shadow-[4px_4px_0px_0px_black] active:translate-y-1 active:shadow-none transition-all">
              Generate Voice
            </button>
          </div>
        )}
      </div>
    </div>
  );

  const ControlsPanel = () => (
    <div className="w-80 bg-white border-l-[4px] border-black flex flex-col shrink-0 z-10">
      <div className="flex-1 p-6 overflow-y-auto space-y-6">
        <div className="bg-[#FFDD00] border-[4px] border-black p-4 rounded-xl shadow-[4px_4px_0px_0px_black]">
          <h2 className="font-black uppercase text-xl tracking-tight flex justify-between items-center">
            Controls
            {activeLayer && activeLayer.keyframes?.length > 0 && <span className="text-[10px] bg-black text-[#CCFF00] px-2 py-1 rounded">KFS: {activeLayer.keyframes.length}</span>}
          </h2>
        </div>

        {!activeLayer || !activeInterpProps ? (
          <div className="text-center text-gray-400 font-bold uppercase p-8 border-4 border-dashed border-gray-300">Select a layer</div>
        ) : (
          <>
            <div className="bg-gray-100 p-4 border-[3px] border-black rounded-xl shadow-[4px_4px_0px_0px_black]">
              <h4 className="font-black uppercase text-sm border-b-[3px] border-black pb-1 mb-3">Transitions Library</h4>
              <label className="text-[10px] font-black uppercase text-gray-500 mb-2 block">Entry</label>
              <div className="grid grid-cols-3 gap-2 mb-4">
                {['none', 'fade', 'pop', 'slideUp', 'glitch'].map(anim => (
                  <button key={`entry-${anim}`} onClick={() => updateLayer(activeLayer.id, { entryAnim: anim as any })} className={`p-1.5 border-[2px] border-black rounded text-[9px] font-black uppercase transition-all ${activeLayer.entryAnim === anim ? 'bg-[#00CCFF] shadow-[2px_2px_0px_0px_black]' : 'bg-white hover:bg-gray-200'}`}>{anim}</button>
                ))}
              </div>
              <label className="text-[10px] font-black uppercase text-gray-500 mb-2 block">Continuous Motion</label>
              <div className="grid grid-cols-2 gap-2">
                {['none', 'zoomIn', 'panRight', 'float'].map(anim => (
                  <button key={`motion-${anim}`} onClick={() => updateLayer(activeLayer.id, { motionAnim: anim as any })} disabled={activeLayer.keyframes?.length > 0} className={`p-1.5 border-[2px] border-black rounded text-[9px] font-black uppercase transition-all disabled:opacity-30 ${activeLayer.motionAnim === anim ? 'bg-[#FF3399] text-white shadow-[2px_2px_0px_0px_black]' : 'bg-white hover:bg-gray-200'}`}>{anim === 'none' ? 'Static / KFS' : anim}</button>
                ))}
              </div>
            </div>

            <div className="bg-gray-100 p-4 border-[3px] border-black rounded-xl shadow-[4px_4px_0px_0px_black]">
              <h4 className="font-black uppercase text-sm border-b-[3px] border-black pb-1 mb-3">Properties</h4>
              {[
                { label: 'X Pos', prop: 'x', min: -500, max: 500, step: 10 },
                { label: 'Y Pos', prop: 'y', min: -500, max: 500, step: 10 },
                { label: 'Scale', prop: 'scale', min: 0.1, max: 3, step: 0.1 },
                { label: 'Rotate', prop: 'rotation', min: -360, max: 360, step: 1 },
                { label: 'Opacity', prop: 'opacity', min: 0, max: 1, step: 0.1 },
                { label: 'Hue', prop: 'hsb.hue', min: 0, max: 360, step: 1 },
                { label: 'Sat', prop: 'hsb.sat', min: 0, max: 200, step: 1 },
              ].map(ctrl => {
                 const isHsb = ctrl.prop.includes('hsb.');
                 const val = isHsb ? activeInterpProps.hsb[ctrl.prop.split('.')[1] as keyof typeof activeInterpProps.hsb] : activeInterpProps[ctrl.prop as keyof typeof activeInterpProps];
                 return (
                  <div key={ctrl.prop} className="mb-3">
                    <div className="flex justify-between items-center text-[10px] font-black uppercase mb-1">
                       <div className="flex items-center gap-1">
                          <button onClick={() => toggleKeyframe(activeLayer.id)} className={`w-3 h-3 rotate-45 border-[2px] border-black transition-colors ${hasKeyframeAtCurrent ? 'bg-[#CCFF00]' : 'bg-white hover:bg-gray-300'}`} />
                          <label>{ctrl.label}</label>
                       </div>
                       <span>{Number(val).toFixed(1)}</span>
                    </div>
                    <input type="range" min={ctrl.min} max={ctrl.max} step={ctrl.step} value={Number(val)} onChange={e => handlePropChange(ctrl.prop, parseFloat(e.target.value))} className="w-full accent-[#00CCFF] h-2 bg-gray-300 rounded-full appearance-none cursor-pointer" />
                  </div>
                 )
              })}
              <div className="flex gap-2 mt-4">
                 <button onClick={() => toggleKeyframe(activeLayer.id)} className="flex-1 py-2 bg-[#CCFF00] border-[2px] border-black font-black uppercase text-[9px] shadow-[2px_2px_0px_0px_black] active:translate-y-1 active:shadow-none">➕ Set KF</button>
                 <button onClick={() => updateLayer(activeLayer.id, { keyframes: [] })} className="flex-1 py-2 bg-black text-white border-[2px] border-black font-black uppercase text-[9px] shadow-[2px_2px_0px_0px_black] active:translate-y-1 active:shadow-none">Clear KFS</button>
              </div>
            </div>
          </>
        )}
      </div>
    </div>
  );

  const TimelineArea = () => (
    <div className="h-full bg-white flex flex-col z-30">
      <div className="h-12 bg-gray-100 border-b-[3px] border-black flex items-center px-4 gap-6 shrink-0 justify-between">
        <div className="flex items-center gap-4">
          <button onClick={() => setIsPlaying(!isPlaying)} className="p-2 bg-black text-white rounded-lg hover:bg-[#FF3399] transition-colors shadow-[2px_2px_0px_0px_rgba(0,0,0,0.3)]">
            {isPlaying ? <Pause size={18} className="fill-white"/> : <Play size={18} className="fill-white"/>}
          </button>
          
          <div className="font-mono font-black text-lg bg-black text-white px-2 py-1 rounded shadow-inner flex items-center gap-1">
            <div className="bg-[#FF3399] px-1 rounded flex items-center shadow-[1px_1px_0px_0px_#000]">
              <input type="number" value={Number(currentTime).toFixed(1)} onChange={e => setCurrentTime(getSnappedTime(Math.min(maxTime, Math.max(0, parseFloat(e.target.value) || 0))))} className="bg-transparent text-white w-12 outline-none text-right" step="0.1" />s
            </div>
            <span className="text-gray-500 text-sm flex items-center ml-1">
              / <input type="number" value={maxTime} onChange={e => setMaxTime(Math.max(1, parseFloat(e.target.value) || 15))} className="bg-transparent text-gray-500 w-8 outline-none text-right ml-1" />s
            </span>
          </div>

          <button onClick={handleCut} title="Split Clip at Playhead" className="p-1.5 bg-white border-[2px] border-black rounded-lg hover:bg-[#CCFF00] shadow-[2px_2px_0px_0px_black] ml-2 font-black uppercase text-[10px] flex items-center gap-1">
            <Scissors size={14} /> Cut
          </button>
        </div>
        
        <div className="flex items-center gap-4">
          <button onClick={() => setIsSnapping(!isSnapping)} className={`flex items-center gap-1 px-3 py-1 border-[2px] border-black rounded-lg text-[10px] font-black uppercase transition-all ${isSnapping ? 'bg-[#CCFF00] shadow-[2px_2px_0px_0px_black]' : 'bg-white opacity-50'}`}>
            <Magnet size={14}/> Snap
          </button>
          <div className="flex items-center gap-2">
            <Search size={14} className="text-gray-500" />
            <input type="range" min="1" max="5" step="0.1" value={timelineZoom} onChange={e => setTimelineZoom(parseFloat(e.target.value))} className="w-24 accent-black h-2 bg-gray-300 rounded-full cursor-pointer" />
          </div>
        </div>
      </div>

      <div className="flex-1 flex overflow-hidden relative">
        <div className="w-56 bg-white border-r-[4px] border-black shrink-0 overflow-y-auto pb-8 z-20">
          {layers.slice().reverse().map(layer => (
            <div key={`head-${layer.id}`} onClick={() => setActiveLayerId(layer.id)} className={`h-12 border-b-[3px] border-black p-2 flex items-center justify-between cursor-pointer transition-colors ${activeLayerId === layer.id ? 'bg-[#CCFF00]' : 'hover:bg-gray-100'}`}>
              <span className="font-black uppercase text-[10px] truncate w-24">{layer.name}</span>
              <div className="flex gap-1">
                <button onClick={(e) => { e.stopPropagation(); updateLayer(layer.id, { isVisible: !layer.isVisible }); }} className="p-1 hover:bg-black hover:text-white rounded">
                  {layer.isVisible ? <Eye size={12}/> : <EyeOff size={12}/>}
                </button>
                <button onClick={(e) => { e.stopPropagation(); deleteLayer(layer.id); }} className="p-1 hover:bg-red-500 hover:text-white rounded">
                  <Trash2 size={12}/>
                </button>
              </div>
            </div>
          ))}
        </div>

        <div className="flex-1 bg-gray-200 relative overflow-y-auto overflow-x-auto pb-8 custom-scrollbar">
          <div className="relative h-full" style={{ minWidth: `${timelineZoom * 100}%` }} ref={trackAreaRef}>
            
            {/* Grid Markers */}
            <div className="absolute inset-0 pointer-events-none opacity-20 flex">
               {[...Array(Math.floor(maxTime))].map((_, i) => (
                 <div key={i} className="flex-1 border-l border-black h-full relative">
                   <span className="absolute top-1 left-1 text-[8px] font-black">{i}s</span>
                 </div>
               ))}
            </div>

            {/* Playhead */}
            <div className="absolute top-0 bottom-0 w-0.5 bg-[#FF3399] z-30 pointer-events-none shadow-[0_0_10px_#FF3399]" style={{ left: `${(currentTime / maxTime) * 100}%` }}>
              <div className="absolute -top-1 left-1/2 -translate-x-1/2 w-3 h-3 bg-[#FF3399] border-2 border-black rotate-45" />
            </div>

            <input type="range" min="0" max={maxTime} step="0.05" value={currentTime} onChange={e => setCurrentTime(getSnappedTime(parseFloat(e.target.value)))} className="absolute top-0 w-full h-full opacity-0 z-40 cursor-col-resize" />

            <div className="relative">
              {layers.slice().reverse().map(layer => (
                <div key={`track-${layer.id}`} className="h-12 border-b-[2px] border-black/20 relative">
                  <div 
                    onClick={() => setActiveLayerId(layer.id)}
                    className={`absolute top-1 bottom-1 border-[3px] border-black rounded flex items-center px-2 cursor-grab active:cursor-grabbing
                    ${activeLayerId === layer.id ? 'bg-[#00CCFF] shadow-[2px_2px_0px_0px_black] z-10' : 'bg-white z-0'}`}
                    style={{ left: `${(layer.start / maxTime) * 100}%`, width: `${((layer.end - layer.start) / maxTime) * 100}%` }}
                    onPointerDown={(e) => handlePointerDown(e, layer.id, 'move')}
                  >
                    <div className="absolute left-0 top-0 bottom-0 w-4 cursor-ew-resize hover:bg-black/20 z-20 flex items-center justify-center border-r-[2px] border-black/10" onPointerDown={(e) => handlePointerDown(e, layer.id, 'start')} />
                    <span className="font-black text-[10px] uppercase truncate pointer-events-none px-2">{layer.name}</span>
                    <div className="absolute right-0 top-0 bottom-0 w-4 cursor-ew-resize hover:bg-black/20 z-20 flex items-center justify-center border-l-[2px] border-black/10" onPointerDown={(e) => handlePointerDown(e, layer.id, 'end')} />
                  </div>
                  
                  {layer.keyframes?.map((kf, idx) => (
                    <div key={`kf-${layer.id}-${idx}`} className="absolute top-1/2 -translate-y-1/2 w-3 h-3 bg-[#CCFF00] border-[2px] border-black rotate-45 z-20 pointer-events-none" style={{ left: `calc(${(kf.time / maxTime) * 100}% - 6px)` }} />
                  ))}
                </div>
              ))}
            </div>
          </div>
        </div>
      </div>
    </div>
  );

  const CanvasArea = () => (
    <div className={`relative bg-black border-[8px] border-black shadow-[24px_24px_0px_0px_rgba(0,0,0,0.5)] overflow-hidden transition-all duration-500 flex items-center justify-center ${aspectRatio === '9:16' ? 'h-[95%] aspect-[9/16]' : 'w-[95%] aspect-[16/9]'}`}>
      
      <div className="absolute top-4 left-1/2 -translate-x-1/2 z-50 flex gap-2 bg-white/20 backdrop-blur-md p-1 border-[2px] border-black rounded-xl">
        <button onClick={() => setAspectRatio('9:16')} className={`p-2 rounded-lg transition-colors ${aspectRatio === '9:16' ? 'bg-[#FF3399] text-white shadow-[2px_2px_0px_0px_black]' : 'text-white/70 hover:text-white'}`}>
          <Smartphone size={20} />
        </button>
        <button onClick={() => setAspectRatio('16:9')} className={`p-2 rounded-lg transition-colors ${aspectRatio === '16:9' ? 'bg-[#00CCFF] text-black shadow-[2px_2px_0px_0px_black]' : 'text-white/70 hover:text-white'}`}>
          <Monitor size={20} />
        </button>
      </div>

      <AnimatePresence>
        {layers.filter(l => l.type !== 'audio' && l.isVisible && currentTime >= l.start && currentTime <= l.end).sort((a,b) => a.zIndex - b.zIndex).map(layer => {
            const styles = getLayerStyles(layer, currentTime, maxTime);

            return (
              <motion.div 
                key={layer.id} 
                animate={{
                  x: styles.x, y: styles.y, scale: styles.scale, rotate: styles.rotate, opacity: styles.opacity,
                  filter: styles.filter
                }}
                transition={{ duration: 0 }} // 100% Deterministic: Instant live tracking for pause previews
                className={`absolute inset-0 flex items-center justify-center ${activeLayerId === layer.id ? 'ring-[6px] ring-[#00CCFF] ring-inset z-[100]' : ''}`} 
                style={{ zIndex: layer.zIndex }}
              >
                {layer.type === 'media' && <img src={layer.url} className="w-full h-full object-cover" alt={layer.name} />}
                {layer.type === 'text' && ( <div className="text-center px-4 uppercase font-black tracking-tighter" style={{ fontSize: 'clamp(3rem, 10vw, 8rem)', lineHeight: 1, color: layer.color || '#FFF', textShadow: '4px 4px 0px #000, -2px -2px 0px #000, 2px -2px 0px #000, -2px 2px 0px #000' }}>{layer.url}</div> )}
              </motion.div>
            )
        })}
      </AnimatePresence>
    </div>
  );

  return (
    <div className="h-screen w-full bg-[#E5E7EB] flex flex-col overflow-hidden font-sans selection:bg-[#FF3399] selection:text-white">
      <style dangerouslySetInnerHTML={{__html: `
        input[type=number]::-webkit-inner-spin-button, input[type=number]::-webkit-outer-spin-button { -webkit-appearance: none; margin: 0; }
        .custom-scrollbar::-webkit-scrollbar { width: 8px; height: 8px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: #f1f1f1; border-left: 2px solid black; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #000; }
      `}} />
      <AnimatePresence>
        {toastMsg && <motion.div initial={{ opacity: 0, y: -50 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: -50 }} className="fixed top-6 left-1/2 -translate-x-1/2 z-[2000] bg-[#CCFF00] text-black border-[3px] border-black px-6 py-2 rounded-xl font-black shadow-[4px_4px_0px_0px_black] uppercase">{toastMsg}</motion.div>}
      </AnimatePresence>

      <ToolHeader title="Shorts Studio Pro" titleBgColor="bg-[#FF3399]" iconBgColor="bg-[#ccff00]" />

      {aspectRatio === '9:16' ? (
        /* --- VERTICAL LAYOUT (9:16) --- */
        /* Media Player goes FULL HEIGHT on the FAR RIGHT. Timeline + Tools take the rest. */
        <div className="flex-1 flex flex-row overflow-hidden border-t-[4px] border-black">
          <VerticalNavMenu />
          <div className="flex-1 flex flex-col overflow-hidden">
            <div className="flex-1 flex overflow-hidden">
               <ToolPanel />
               <ControlsPanel />
            </div>
            <div className="h-[40%] min-h-[280px] border-t-[4px] border-black shrink-0">
               <TimelineArea />
            </div>
          </div>
          <div className="w-[450px] xl:w-[550px] shrink-0 border-l-[4px] border-black bg-[url('https://www.transparenttextures.com/patterns/cubes.png')] shadow-inner flex items-center justify-center p-8 relative">
             <CanvasArea />
          </div>
        </div>
      ) : (
        /* --- LANDSCAPE LAYOUT (16:9) --- */
        /* Timeline spans absolute FULL WIDTH underneath everything. Canvas Top Right/Middle. */
        <div className="flex-1 flex flex-col overflow-hidden border-t-[4px] border-black">
          <div className="flex-1 flex flex-row overflow-hidden">
             <VerticalNavMenu />
             <ToolPanel />
             <ControlsPanel />
             <div className="flex-1 bg-[url('https://www.transparenttextures.com/patterns/cubes.png')] shadow-inner flex items-center justify-center p-8 relative border-l-[4px] border-black">
                <CanvasArea />
             </div>
          </div>
          <div className="h-[40%] min-h-[280px] border-t-[4px] border-black shrink-0 w-full">
             <TimelineArea />
          </div>
        </div>
      )}
    </div>
  );
};

export default function App() { return <ShortsStudio />; }
