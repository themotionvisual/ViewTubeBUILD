const pcmToWav = (pcmData: string, sampleRate: number) => \{\
  const binaryString = atob(pcmData);\
  const bytes = new Uint8Array(binaryString.length);\
  for (let i = 0; i < binaryString.length; i++) bytes[i] = binaryString.charCodeAt(i);\
  const wavBuffer = new ArrayBuffer(44 + bytes.length);\
  const view = new DataView(wavBuffer);\
  const writeString = (v: DataView, o: number, s: string) => \{ for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); \};\
  writeString(view, 0, 'RIFF'); view.setUint32(4, 36 + bytes.length, true); writeString(view, 8, 'WAVE');\
  writeString(view, 12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);\
  view.setUint16(22, 1, true); view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);\
  view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);\
  new Uint8Array(wavBuffer, 44).set(bytes);\

const pcmToWav = (pcmData: string, sampleRate: number) => \{\
  const binaryString = atob(pcmData);\
  const bytes = new Uint8Array(binaryString.length);\
  for (let i = 0; i < binaryString.length; i++) bytes[i] = binaryString.charCodeAt(i);\
  const wavBuffer = new ArrayBuffer(44 + bytes.length);\
  const view = new DataView(wavBuffer);\
  const writeString = (v: DataView, o: number, s: string) => \{ for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); \};\
  writeString(view, 0, 'RIFF'); view.setUint32(4, 36 + bytes.length, true); writeString(view, 8, 'WAVE');\
  writeString(view, 12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);\
  view.setUint16(22, 1, true); view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);\
  view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);\
  new Uint8Array(wavBuffer, 44).set(bytes);\
  return URL.createObjectURL(new Blob([wavBuffer], \{ type: 'audio/wav' \}));\

const getLayerStyles = (layer: Layer, time: number, maxTime: number) => \{\
  let props = \{\
    x: layer.x, y: layer.y, scale: layer.scale, rotate: layer.rotation, opacity: layer.opacity,\
    filter: `hue-rotate($\{layer.hsb.hue\}deg) saturate($\{layer.hsb.sat\}%) brightness($\{layer.hsb.bri\}%) blur($\{layer.blur\}px)`\
  \};\
  \
  if (layer.keyframes && layer.keyframes.length > 0) \{\
    const kfs = [...layer.keyframes].sort((a, b) => a.time - b.time);\
    const prev = kfs.slice().reverse().find(k => k.time <= time);\
    const next = kfs.find(k => k.time > time);\
\
    if (prev && !next) \{\
      props = \{ ...props, x: prev.props.x, y: prev.props.y, scale: prev.props.scale, rotate: prev.props.rotation, opacity: prev.props.opacity, filter: `hue-rotate($\{prev.props.hsb.hue\}deg) saturate($\{prev.props.hsb.sat\}%) brightness($\{prev.props.hsb.bri\}%) blur($\{prev.props.blur\}px)` \};\
    \} else if (prev && next) \{\
      const t = (time - prev.time) / (next.time - prev.time);\
      props = \{\
        ...props,\
        x: lerp(prev.props.x, next.props.x, t), y: lerp(prev.props.y, next.props.y, t),\
        scale: lerp(prev.props.scale, next.props.scale, t), rotate: lerp(prev.props.rotation, next.props.rotation, t),\
        opacity: lerp(prev.props.opacity, next.props.opacity, t),\
        filter: `hue-rotate($\{lerp(prev.props.hsb.hue, next.props.hsb.hue, t)\}deg) saturate($\{lerp(prev.props.hsb.sat, next.props.hsb.sat, t)\}%) brightness($\{lerp(prev.props.hsb.bri, next.props.hsb.bri, t)\}%) blur($\{lerp(prev.props.blur, next.props.blur, t)\}px)`\
      \};\
    \}\
  \}\
\
  const timeInClip = time - layer.start;\
  const dur = layer.end - layer.start;\
  if (timeInClip < 0) return props; // Handled by AnimatePresence mounting\
\
  // Entry Animations\
  const entryDur = Math.min(0.5, dur);\
  const ep = Math.min(1, Math.max(0, timeInClip / entryDur));\
  const e = 1 - Math.pow(1 - ep, 3); // easeOutCubic\
\
  if (layer.entryAnim === 'fade') props.opacity *= e;\
  if (layer.entryAnim === 'pop') \{ props.scale *= (0.5 + 0.5 * e); props.opacity *= e; \}\
  if (layer.entryAnim === 'slideUp') \{ props.y += 100 * (1 - e); props.opacity *= e; \}\
  if (layer.entryAnim === 'glitch' && e < 1) \{ props.x += (Math.random() * 20 - 10) * (1 - e); props.opacity *= e; \}\

  // Continuous Motion\
  if (layer.motionAnim !== 'none' && (!layer.keyframes || layer.keyframes.length === 0)) \{\
    const mp = timeInClip / maxTime;\
    if (layer.motionAnim === 'panRight') props.x += 100 * mp;\
    if (layer.motionAnim === 'zoomIn') props.scale *= (1 + 0.2 * mp);\
    if (layer.motionAnim === 'float') props.y += Math.sin(timeInClip * 2) * 20;\
  \}\
\
  return props;\

const pcmToWav = (pcmData: string, sampleRate: number) => \{\
  const binaryString = atob(pcmData);\
  const bytes = new Uint8Array(binaryString.length);\
  for (let i = 0; i < binaryString.length; i++) bytes[i] = binaryString.charCodeAt(i);\
  const wavBuffer = new ArrayBuffer(44 + bytes.length);\
  const view = new DataView(wavBuffer);\
  const writeString = (v: DataView, o: number, s: string) => \{ for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); \};\
  writeString(view, 0, 'RIFF'); view.setUint32(4, 36 + bytes.length, true); writeString(view, 8, 'WAVE');\
  writeString(view, 12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);\
  view.setUint16(22, 1, true); view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);\
  view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);\
  new Uint8Array(wavBuffer, 44).set(bytes);\
  return URL.createObjectURL(new Blob([wavBuffer], \{ type: 'audio/wav' \}));\

type LayerType = 'media' | 'text' | 'audio';\

interface Keyframe \{\
  time: number;\
  props: \{ x: number; y: number; scale: number; rotation: number; opacity: number; blur: number; hsb: \{ hue: number; sat: number; bri: number \} \};\

interface Layer \{\
  id: string; name: string; url: string; type: LayerType; isVisible: boolean; color?: string;\
  start: number; end: number;\
  x: number; y: number; scale: number; rotation: number; opacity: number; blur: number; hsb: \{ hue: number; sat: number; bri: number \};\
  entryAnim: 'none' | 'fade' | 'pop' | 'slideUp' | 'glitch';\

const lerp = (start: number, end: number, t: number) => start + (end - start) * t;\
\
// Gets the exact mathematical properties of a layer at a given time\
const getInterpolatedProps = (layer: Layer, time: number) => \{\
  const baseProps = \{ x: layer.x, y: layer.y, scale: layer.scale, rotation: layer.rotation, opacity: layer.opacity, blur: layer.blur, hsb: \{ ...layer.hsb \} \};\
  if (!layer.keyframes || layer.keyframes.length === 0) return baseProps;\
\
  const kfs = [...layer.keyframes].sort((a, b) => a.time - b.time);\
  const prev = kfs.slice().reverse().find(k => k.time <= time);\
  const next = kfs.find(k => k.time > time);\
\
  if (prev && !next) return JSON.parse(JSON.stringify(prev.props));\
  if (!prev && next) return baseProps;\
  if (prev && next) \{\
    const t = (time - prev.time) / (next.time - prev.time);\
    return \{\
      x: lerp(prev.props.x, next.props.x, t), y: lerp(prev.props.y, next.props.y, t),\
      scale: lerp(prev.props.scale, next.props.scale, t), rotation: lerp(prev.props.rotation, next.props.rotation, t),\
      opacity: lerp(prev.props.opacity, next.props.opacity, t), blur: lerp(prev.props.blur, next.props.blur, t),\
      hsb: \{ hue: lerp(prev.props.hsb.hue, next.props.hsb.hue, t), sat: lerp(prev.props.hsb.sat, next.props.hsb.sat, t), bri: lerp(prev.props.hsb.bri, next.props.hsb.bri, t) \}\
    \};\
  \}\
  return baseProps;\
\};\

// Returns the final styling for the canvas based on interpolation + entry/motion animations\
const getLayerStyles = (layer: Layer, time: number, maxTime: number) => \{\
  const interp = getInterpolatedProps(layer, time);\
  \
  let props = \{\
    x: interp.x, y: interp.y, scale: interp.scale, rotate: interp.rotation, opacity: interp.opacity,\
    filter: `hue-rotate($\{interp.hsb.hue\}deg) saturate($\{interp.hsb.sat\}%) brightness($\{interp.hsb.bri\}%) blur($\{interp.blur\}px)`\
  \};\
\
  const timeInClip = time - layer.start;\
  const dur = layer.end - layer.start;\
  if (timeInClip < 0) return props; // Handled by AnimatePresence mounting\
\
  // Entry Animations\
  const entryDur = Math.min(0.5, dur);\
  const ep = Math.min(1, Math.max(0, timeInClip / entryDur));\
  const e = 1 - Math.pow(1 - ep, 3); // easeOutCubic\
\
  if (layer.entryAnim === 'fade') props.opacity *= e;\
  if (layer.entryAnim === 'pop') \{ props.scale *= (0.5 + 0.5 * e); props.opacity *= e; \}\
  if (layer.entryAnim === 'slideUp') \{ props.y += 100 * (1 - e); props.opacity *= e; \}\
  if (layer.entryAnim === 'glitch' && e < 1) \{ props.x += (Math.random() * 20 - 10) * (1 - e); props.opacity *= e; \}\
\
  // Continuous Motion\
  if (layer.motionAnim !== 'none' && (!layer.keyframes || layer.keyframes.length === 0)) \{\
    const mp = timeInClip / maxTime;\
    if (layer.motionAnim === 'panRight') props.x += 100 * mp;\
    if (layer.motionAnim === 'zoomIn') props.scale *= (1 + 0.2 * mp);\
    if (layer.motionAnim === 'float') props.y += Math.sin(timeInClip * 2) * 20;\
  \}\
\
  return props;\
\};\

const pcmToWav = (pcmData: string, sampleRate: number) => \{\
  const binaryString = atob(pcmData);\
  const bytes = new Uint8Array(binaryString.length);\
  for (let i = 0; i < binaryString.length; i++) bytes[i] = binaryString.charCodeAt(i);\
  const wavBuffer = new ArrayBuffer(44 + bytes.length);\
  const view = new DataView(wavBuffer);\
  const writeString = (v: DataView, o: number, s: string) => \{ for (let i = 0; i < s.length; i++) v.setUint8(o + i, s.charCodeAt(i)); \};\
  writeString(view, 0, 'RIFF'); view.setUint32(4, 36 + bytes.length, true); writeString(view, 8, 'WAVE');\
  writeString(view, 12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);\
  view.setUint16(22, 1, true); view.setUint32(24, sampleRate, true); view.setUint32(28, sampleRate * 2, true);\
  view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(view, 36, 'data'); view.setUint32(40, bytes.length, true);\
  new Uint8Array(wavBuffer, 44).set(bytes);\
  return URL.createObjectURL(new Blob([wavBuffer], \{ type: 'audio/wav' \}));\

const getLayerStyles = (layer: Layer, time: number, maxTime: number) => \{\
  let props = \{\
    x: layer.x, y: layer.y, scale: layer.scale, rotate: layer.rotation, opacity: layer.opacity,\
    filter: `hue-rotate($\{layer.hsb.hue\}deg) saturate($\{layer.hsb.sat\}%) brightness($\{layer.hsb.bri\}%) blur($\{layer.blur\}px)`\
  \};\
  \
  if (layer.keyframes && layer.keyframes.length > 0) \{\
    const kfs = [...layer.keyframes].sort((a, b) => a.time - b.time);\
    const prev = kfs.slice().reverse().find(k => k.time <= time);\
    const next = kfs.find(k => k.time > time);\
\
    if (prev && !next) \{\
      props = \{ ...props, x: prev.props.x, y: prev.props.y, scale: prev.props.scale, rotate: prev.props.rotation, opacity: prev.props.opacity, filter: `hue-rotate($\{prev.props.hsb.hue\}deg) saturate($\{prev.props.hsb.sat\}%) brightness($\{prev.props.hsb.bri\}%) blur($\{prev.props.blur\}px)` \};\
    \} else if (prev && next) \{\
      const t = (time - prev.time) / (next.time - prev.time);\
      props = \{\
        ...props,\
        x: lerp(prev.props.x, next.props.x, t), y: lerp(prev.props.y, next.props.y, t),\
        scale: lerp(prev.props.scale, next.props.scale, t), rotate: lerp(prev.props.rotation, next.props.rotation, t),\
        opacity: lerp(prev.props.opacity, next.props.opacity, t),\
        filter: `hue-rotate($\{lerp(prev.props.hsb.hue, next.props.hsb.hue, t)\}deg) saturate($\{lerp(prev.props.hsb.sat, next.props.hsb.sat, t)\}%) brightness($\{lerp(prev.props.hsb.bri, next.props.hsb.bri, t)\}%) blur($\{lerp(prev.props.blur, next.props.blur, t)\}px)`\
      \};\
    \}\
  \}\
\
  const timeInClip = time - layer.start;\
  const dur = layer.end - layer.start;\
  if (timeInClip < 0) return props; // Handled by AnimatePresence mounting\
\
  // Entry Animations\
  const entryDur = Math.min(0.5, dur);\
  const ep = Math.min(1, Math.max(0, timeInClip / entryDur));\
  const e = 1 - Math.pow(1 - ep, 3); // easeOutCubic\
\
  if (layer.entryAnim === 'fade') props.opacity *= e;\
  if (layer.entryAnim === 'pop') \{ props.scale *= (0.5 + 0.5 * e); props.opacity *= e; \}\
  if (layer.entryAnim === 'slideUp') \{ props.y += 100 * (1 - e); props.opacity *= e; \}\
  if (layer.entryAnim === 'glitch' && e < 1) \{ props.x += (Math.random() * 20 - 10) * (1 - e); props.opacity *= e; \}\

  // Continuous Motion\
  if (layer.motionAnim !== 'none' && (!layer.keyframes || layer.keyframes.length === 0)) \{\
    const mp = timeInClip / maxTime;\
    if (layer.motionAnim === 'panRight') props.x += 100 * mp;\
    if (layer.motionAnim === 'zoomIn') props.scale *= (1 + 0.2 * mp);\
    if (layer.motionAnim === 'float') props.y += Math.sin(timeInClip * 2) * 20;\
  \}\
\
  return props;\
