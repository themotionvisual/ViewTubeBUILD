import cv2
import numpy as np

# Load the image
img = cv2.imread('input_file_5.png')
img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

# The image has large "pixels". Let's find the grid size.
# A simple way is to find edges or just resize down if we can guess the size.
# Let's crop to the whale and find the block size.
# For now, let's just print the unique colors to identify them.
colors = np.unique(img.reshape(-1, img.shape[2]), axis=0)
print("Colors:", colors)

# Let's try to auto-detect the grid size by looking at transitions
gray = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
edges = cv2.Canny(gray, 50, 150)
coords = np.column_stack(np.where(edges > 0))
