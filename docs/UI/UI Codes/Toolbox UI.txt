// ==========================================
// UI COMPONENT: SubToolbox
// Features the stationary top-anchor logic.
// ==========================================
const SubToolbox = ({ title, icon: Icon, headerBg, iconBg, isOpen, onToggle, openUnits = 3, children, config = {} }) => {
  const unit = config.baseHeight || 60;
  const gap = config.gap || 24;
  const stroke = config.stroke || 4;
  
  // Total height = (Units * UnitHeight) + ((Units-1) * Gap)
  const innerHeight = (openUnits * unit) + ((openUnits - 1) * gap) - (56 + (stroke * 2));

  return (
    <div 
      className="w-full bg-white overflow-hidden relative flex flex-col shrink-0 transition-all duration-300"
      style={{ 
        border: `${stroke}px solid black`,
        borderRadius: config.radius !== undefined ? `${config.radius}px` : '16px'
      }}
    >
      <header 
        onClick={onToggle}
        className="h-[56px] min-h-[56px] flex items-center justify-between select-none relative z-20 group transition-[margin] duration-500 ease-[cubic-bezier(0.4,0,0.2,1)] cursor-pointer"
        style={{ 
          backgroundColor: headerBg, 
          color: 'black',
          borderBottom: `${stroke}px solid black`,
          marginBottom: isOpen ? '0px' : `-${stroke}px`
        }}
      >
        <div className="flex items-center h-full flex-1">
          <div 
            className="h-full aspect-square flex items-center justify-center shrink-0"
            style={{ backgroundColor: iconBg, borderRight: `${stroke}px solid black` }}
          >
            <Icon size={32} strokeWidth={2.5} className="text-black" />
          </div>
          <div className="flex flex-col pl-4 justify-center pointer-events-none select-none">
            <h3 className="font-[900] uppercase tracking-tighter text-black leading-none mt-0.5" style={{ fontSize: config.textSize ? `${config.textSize}px` : '32px' }}>
              {title}
            </h3>
          </div>
        </div>
        <div className="flex items-center gap-3 pr-4 h-full">
          <div className="h-full px-2 flex items-center justify-center">
            <AnimatedToggleIcon open={isOpen} size={36} />
          </div>
        </div>
      </header>

      <div className={`grid transition-[grid-template-rows] duration-500 ease-[cubic-bezier(0.4,0,0.2,1)] ${isOpen ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`}>
        <div className="overflow-hidden min-h-0">
          <main 
            className="bg-white w-full p-6 text-black flex flex-col justify-between"
            style={{ height: `${innerHeight}px` }}
          >
            {children}
          </main>
        </div>
      </div>
    </div>
  );
};
