// --- MOCKED DEPENDENCIES ---

const getToolboxPaletteColors = (index: number) => {

  const palettes = [

    { header: '#FFDD00', icon: '#FF3399' },

    { header: '#00CCFF', icon: '#FF9900' },

    { header: '#CCFF00', icon: '#9900FF' },

    { header: '#FF3399', icon: '#00CCFF' },

  ];

  return palettes[index % palettes.length];

};



// CustomIcon: Exact paths for Outward/Inward arrows

const CustomIcon = ({ name, size = 24, className = '' }: { name: string, size?: number, className?: string }) => {

  if (name === 'SYMBOLS 22') {

    return (

      <svg width={size} height={size} viewBox="0 0 24 24" fill="currentColor" className={className}>

        <path d="M12 0 L8 4 H11 V10 H13 V4 H16 Z" />

        <path d="M24 12 L20 8 V11 H14 V13 H20 V16 Z" />

        <path d="M12 24 L8 20 H11 V14 H13 V20 H16 Z" />

        <path d="M0 12 L4 8 V11 H10 V13 H4 V16 Z" />

      </svg>

    );

  }

  if (name === 'SYMBOLS 23') {

    return (

      <svg width={size} height={size} viewBox="0 0 24 24" fill="currentColor" className={className}>

        <path d="M12 11 L8 7 H11 V0 H13 V7 H16 Z" />

        <path d="M13 12 L17 8 V11 H24 V13 H17 V16 Z" />

        <path d="M12 13 L16 17 H13 V24 H11 V17 H8 Z" />

        <path d="M11 12 L7 16 V13 H0 V11 H7 V8 Z" />

      </svg>

    );

  }

  return <Zap size={size} className={className} />;

};



// --- CUSTOM ANIMATED TOGGLE ICON ---

      style={{ width: size, height: size }}

    >

      <div className={`w-full h-full relative transition-transform duration-700 ease-in-out ${open ? 'rotate-180' : 'rotate-0'}`}>

        <CustomIcon

          className={`absolute inset-0 w-full h-full text-black transition-opacity duration-500 ease-in-out ${open ? 'opacity-0' : 'opacity-80 group-hover:opacity-100'}`} 

        />

        <CustomIcon 

          name="SYMBOLS 23"

    </div>

  );

};



// --- TYPES ---

export type ToolboxVariant = 'scaffold' | 'accordion' | 'header';

export type ToolboxIndicator = 'symbols' | 'plusminus' | 'none';



interface ToolboxProps {

  variant?: ToolboxVariant;

  title: string;

  subtitle?: string;

  icon?: React.ReactNode;

  headerColor: string;

  iconBoxColor?: string;

  textColor?: string;

  paletteIndex?: number;

  collapsible?: boolean;

  isOpen?: boolean;

  isOpenInitial?: boolean;

  onToggle?: () => void;

  unmountWhenClosed?: boolean;

  headerActions?: React.ReactNode;

  indicator?: ToolboxIndicator;

  outerClassName?: string;

  shellClassName?: string;

  contentClassName?: string;

  embedded?: boolean;

  children?: React.ReactNode;

  helpTitle?: string;

  helpText?: string;

  helpGuide?: string[];

}



// --- MAIN TOOLBOX COMPONENT ---

export const Toolbox: React.FC<ToolboxProps> = ({

  variant = 'scaffold',

  title,

  subtitle,

  icon,

  headerColor,

  iconBoxColor = 'bg-white',

  textColor = 'text-black',

  paletteIndex,

  collapsible = false,

  isOpen,

  isOpenInitial = true,

  onToggle,

  unmountWhenClosed = true,

  headerActions,

  indicator = 'symbols',

  outerClassName = '',

  shellClassName = '',

  contentClassName = '',

  embedded = false,

  children,

  helpTitle,

  helpText,

  helpGuide,

}) => {

  const [internalOpen, setInternalOpen] = useState(isOpenInitial);

  const [helpOpen, setHelpOpen] = useState(false);

  const [subtitleOpen, setSubtitleOpen] = useState(true);

  const helpIdRef = useRef(`help-${Math.random().toString(36).slice(2)}`);



  useEffect(() => {

    const handler = (e: Event) => {

      const detail = (e as CustomEvent<string>).detail;

      if (detail !== helpIdRef.current) {

        setHelpOpen(false);

      }

    };

    window.addEventListener('toolbox_help_open', handler as EventListener);

    return () => window.removeEventListener('toolbox_help_open', handler as EventListener);

  }, []);



  const controlled = typeof isOpen === 'boolean';

  const open = controlled ? Boolean(isOpen) : internalOpen;



  useEffect(() => {

    if (!open) {

      setHelpOpen(false);

      setSubtitleOpen(true);

    }

  }, [open]);



  const setOpen = () => {

    if (onToggle) {

      onToggle();

      return;

    }

    if (!controlled) setInternalOpen((prev) => !prev);

  };



  const isCollapsible = collapsible || indicator === 'plusminus' || indicator === 'symbols';

  const palette = paletteIndex !== undefined && paletteIndex !== null ? getToolboxPaletteColors(paletteIndex) : null;

  const headerStyle = palette ? { backgroundColor: palette.header } : undefined;

  const iconStyle = palette ? { backgroundColor: palette.icon } : undefined;



  const finalContentClass = useMemo(() => {

    if (contentClassName) return contentClassName;

    if (variant === 'accordion') return 'p-6 bg-white text-black';

    return embedded ? 'p-0' : 'p-8';

  }, [contentClassName, embedded, variant]);



  if (variant === 'header') {

    return (

      <header

        className={`${headerColor} ${textColor} h-[80px] flex items-center justify-between px-0 overflow-hidden border-b-[5px] border-black rounded-t-2xl mb-0 select-none ${outerClassName}`}

        style={headerStyle}

      >

        <div className="flex items-center h-full">

          <div

            className={`${iconBoxColor} h-full w-[80px] flex items-center justify-center border-r-[5px] border-black flex-shrink-0`}

            style={iconStyle}

          >

            {icon}

          </div>

          <h1 className="text-[50px] font-[1000] uppercase tracking-tighter pl-8 leading-none mt-1 select-none pointer-events-none">{title}</h1>

        </div>

        <div className="flex items-center gap-6 pr-6">{headerActions}</div>

      </header>

    );

  }



  // Strict Shadows: Only on Main Scaffolds

  const frameClass =

    variant === 'accordion'

      ? 'w-full border-[4px] border-black rounded-2xl bg-white overflow-hidden relative shadow-none'

      : 'w-full bg-white border-[6px] border-black rounded-2xl shadow-[12px_12px_0px_0px_black] flex flex-col overflow-hidden relative';



  const headerBorderThickness = variant === 'accordion' ? 'border-b-[4px]' : 'border-b-[6px]';

  const closedMargin = variant === 'accordion' ? 'mb-[-4px]' : 'mb-[-6px]';

  

  // Permanent overlapping border animation. Hides the inner line when closed via exact negative margin.

  const headerClass =

    variant === 'accordion'

      ? `${headerColor} ${textColor} h-14 flex items-center justify-between select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open ? closedMargin : 'mb-0'}`

      : `${headerColor} ${textColor} h-[80px] flex items-center justify-between overflow-hidden select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open && !subtitle && !helpText ? closedMargin : 'mb-0'}`;



  const contentAnimClass =

    variant === 'accordion'

      ? `grid transition-[grid-template-rows] duration-500 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`

      : `grid transition-[grid-template-rows] duration-700 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`;



  return (

    <div className={`w-full ${shellClassName} ${outerClassName}`}>

      <div className={frameClass}>

        <header className={headerClass} style={headerStyle}>

          <div

            onClick={isCollapsible ? setOpen : undefined}

            className={`flex items-center h-full flex-1 ${isCollapsible ? 'cursor-pointer' : ''}`}

          >

            <div

              className={`${iconBoxColor} ${

                variant === 'accordion' ? 'h-full aspect-square border-r-[4px]' : 'h-full w-[80px] border-r-[6px]'

              } flex items-center justify-center border-black flex-shrink-0`}

              style={iconStyle}

            >

              {icon}

            </div>



            <div className={`flex flex-col ${variant === 'accordion' ? 'pl-4 justify-center' : 'pl-8'} pointer-events-none select-none`}>

              {variant === 'accordion' ? (

                <h3 className="text-[32px] font-[900] uppercase tracking-tighter leading-none mt-0.5">{title}</h3>

              ) : (

                <h1 className="text-[50px] font-[1000] uppercase tracking-tighter leading-none mt-1 sm:text-[40px] md:text-[50px] truncate">{title}</h1>

              )}

            </div>

          </div>



          <div className={`flex items-center ${variant === 'accordion' ? 'gap-3 pr-4' : 'gap-4 pr-6'} h-full`}>

            {headerActions}

            {indicator === 'plusminus' && (

              <div onClick={setOpen} className="h-full px-2 flex items-center justify-center cursor-pointer">

                <span className="text-2xl font-black leading-none text-black">{open ? '−' : '+'}</span>

              </div>

            )}

            {indicator === 'symbols' && isCollapsible && (

              <div onClick={setOpen} className="flex items-center justify-center">

                <AnimatedToggleIcon open={open} size={variant === 'accordion' ? 36 : 48} />

              </div>

            )}

          </div>

        </header>



        <div className={contentAnimClass}>

          <div className="overflow-hidden min-h-0">

            {(subtitle || helpText) && (

              <div

                className={`bg-white border-b-[5px] border-black px-6 ${

                  subtitleOpen ? (helpOpen ? 'py-4' : 'py-3') : 'py-1'

                } transition-all`}

              >

                <div className="flex items-center justify-between gap-4">

                  <div className="text-[11px] font-black uppercase tracking-widest text-black/70">

                    {subtitleOpen ? subtitle : null}

                  </div>

                  {helpText ? (

                    <div className="flex items-center gap-2">

                      <button

                        type="button"

                        onClick={(e) => {

                          e.stopPropagation();

                          const next = !helpOpen;

                          if (next) window.dispatchEvent(new CustomEvent('toolbox_help_open', { detail: helpIdRef.current }));

                          setHelpOpen(next);

                          if (!subtitleOpen) setSubtitleOpen(true);

                        }}

                        className="w-8 h-8 rounded-full border-[3px] border-black text-black bg-white font-black flex items-center justify-center leading-none hover:bg-black hover:text-white transition-all shadow-[2px_2px_0px_0px_black] active:shadow-none active:translate-y-0.5"

                      >

                        ?

                      </button>

                      <button

                        type="button"

                        onClick={() => {

                          const next = !subtitleOpen;

                          setSubtitleOpen(next);

                          if (!next) setHelpOpen(false);

                        }}

                        className="w-8 h-8 rounded-full border-[3px] border-black text-black bg-white font-black flex items-center justify-center leading-none hover:bg-black hover:text-white transition-all shadow-[2px_2px_0px_0px_black] active:shadow-none active:translate-y-0.5"

                      >

                        {subtitleOpen ? '−' : '+'}

                      </button>

                    </div>

                  ) : (

                    <div className="w-8 h-8" />

                  )}

                </div>

                {helpText && helpOpen && subtitleOpen && (

                  <div className="flex flex-col gap-3 mt-4">

                    <div className="flex items-start justify-between gap-4">

                      <div>

                        <h4 className="text-lg font-[1000] uppercase tracking-tighter text-black">

                          {helpTitle || title}

                        </h4>

                        <p className="text-xs font-bold uppercase tracking-widest text-black/50 mt-1">

                          Quick Guide

                        </p>

                      </div>

                      <button

                        type="button"

                        onClick={() => setHelpOpen(false)}

                        className="w-8 h-8 border-[3px] border-black rounded-full flex items-center justify-center font-black hover:bg-black hover:text-white transition-colors"

                      >

                        ×

                      </button>

                    </div>

                    <p className="text-sm font-bold leading-relaxed text-black">

                      {helpText}

                    </p>

                  </div>

                )}

              </div>

            )}

            

            {(!unmountWhenClosed || open) && (

              <main className={`bg-white w-full ${finalContentClass}`}>{children}</main>

            )}

          </div>

        </div>

      </div>

    </div>

  );

};

// --- TYPES ---

export type ToolboxVariant = 'scaffold' | 'accordion' | 'header';

export type ToolboxIndicator = 'symbols' | 'plusminus' | 'none';



interface ToolboxProps {

  variant?: ToolboxVariant;

  title: string;

  subtitle?: string;

  icon?: React.ReactNode;

  headerColor: string;

  iconBoxColor?: string;

  textColor?: string;

  collapsible?: boolean;

  isOpen?: boolean;

  isOpenInitial?: boolean;

  onToggle?: () => void;

  unmountWhenClosed?: boolean;

  headerActions?: React.ReactNode;

  indicator?: ToolboxIndicator;

  outerClassName?: string;

  shellClassName?: string;

  contentClassName?: string;

  embedded?: boolean;

  children?: React.ReactNode;

  helpTitle?: string;

  helpText?: string;

}



// --- MAIN TOOLBOX COMPONENT ---

export const Toolbox: React.FC<ToolboxProps> = ({

  variant = 'scaffold',

  title,

  subtitle,

  icon,

  headerColor,

  iconBoxColor = 'bg-white',

  textColor = 'text-black',

  collapsible = false,

  isOpen,

  isOpenInitial = true,

  onToggle,

  unmountWhenClosed = true,

  headerActions,

  indicator = 'symbols',

  outerClassName = '',

  shellClassName = '',

  contentClassName = '',

  embedded = false,

  children,

  helpTitle,

  helpText,

}) => {

  const [internalOpen, setInternalOpen] = useState(isOpenInitial);

  const [helpOpen, setHelpOpen] = useState(false);

  const [subtitleOpen, setSubtitleOpen] = useState(true);

  const helpIdRef = useRef(`help-${Math.random().toString(36).slice(2)}`);



  useEffect(() => {

    const handler = (e: Event) => {

      const detail = (e as CustomEvent<string>).detail;

      if (detail !== helpIdRef.current) {

        setHelpOpen(false);

      }

    };

    window.addEventListener('toolbox_help_open', handler as EventListener);

    return () => window.removeEventListener('toolbox_help_open', handler as EventListener);

  }, []);



  const controlled = typeof isOpen === 'boolean';

  const open = controlled ? Boolean(isOpen) : internalOpen;



  useEffect(() => {

    if (!open) {

      setHelpOpen(false);

      setSubtitleOpen(true);

    }

  }, [open]);



  const setOpen = () => {

    if (onToggle) {

      onToggle();

      return;

    }

    if (!controlled) setInternalOpen((prev) => !prev);

  };



  const isCollapsible = collapsible || indicator === 'plusminus' || indicator === 'symbols';

  const headerStyle = { backgroundColor: headerColor };

  const iconStyle = { backgroundColor: iconBoxColor };



  const finalContentClass = useMemo(() => {

    if (contentClassName) return contentClassName;

    if (variant === 'accordion') return 'p-6 bg-white text-black';

    return embedded ? 'p-0' : 'p-8';

  }, [contentClassName, embedded, variant]);



  if (variant === 'header') return null;



  // Strict Shadows: Scaffolds cast shadows. Accordions DO NOT.

  const frameClass =

    variant === 'accordion'

      ? 'w-full border-[4px] border-black rounded-2xl bg-white overflow-hidden relative shadow-none'

      : 'w-full bg-white border-[6px] border-black rounded-2xl 12px_0px_0px_black] fshadow-[12px_lex flex-col overflow-hidden relative';



  const headerBorderThickness = variant === 'accordion' ? 'border-b-[4px]' : 'border-b-[6px]';

  const closedMargin = variant === 'accordion' ? 'mb-[-4px]' : 'mb-[-6px]';

  

  // Permanent negative margin overlaps the bottom border perfectly when closed

  const headerClass =

    variant === 'accordion'

      ? `${headerColor} ${textColor} h-14 flex items-center justify-between select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open ? closedMargin : 'mb-0'}`

      : `${headerColor} ${textColor} h-[80px] flex items-center justify-between overflow-hidden select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open && !subtitle && !helpText ? closedMargin : 'mb-0'}`;



  const contentAnimClass =

    variant === 'accordion'

      ? `grid transition-[grid-template-rows] duration-500 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`

      : `grid transition-[grid-template-rows] duration-700 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`;



  return (

    <div className={`w-full h-full flex flex-col ${shellClassName} ${outerClassName}`}>

      <div className={`${frameClass} h-full`}>

        <header className={headerClass} style={headerStyle}>

          <div

            onClick={isCollapsible ? setOpen : undefined}

            className={`flex items-center h-full flex-1 ${isCollapsible ? 'cursor-pointer' : ''}`}

          >

            <div

              className={`${variant === 'accordion' ? 'h-full aspect-square border-r-[4px]' : 'h-full w-[80px] border-r-[6px]'} flex items-center justify-center border-black flex-shrink-0`}

              style={iconStyle}

            >

              {icon}

            </div>



            <div className={`flex flex-col ${variant === 'accordion' ? 'pl-4 justify-center' : 'pl-8'} pointer-events-none select-none`}>

              {variant === 'accordion' ? (

                <h3 className="text-[32px] font-[900] uppercase tracking-tighter leading-none mt-0.5">{title}</h3>

              ) : (

                <h1 className="text-[50px] font-[1000] uppercase tracking-tighter leading-none mt-1 sm:text-[40px] md:text-[50px] truncate">{title}</h1>

              )}

            </div>

          </div>



          <div className={`flex items-center ${variant === 'accordion' ? 'gap-3 pr-4' : 'gap-4 pr-6'} h-full`}>

            {headerActions}

            {indicator === 'plusminus' && (

              <div onClick={setOpen} className="h-full px-2 flex items-center justify-center cursor-pointer">

                <span className="text-2xl font-black leading-none text-black">{open ? '−' : '+'}</span>

              </div>

            )}

            {indicator === 'symbols' && isCollapsible && (

              <div onClick={setOpen} className="flex items-center justify-center h-full">

                <AnimatedToggleIcon open={open} size={variant === 'accordion' ? 36 : 48} />

              </div>

            )}

          </div>

        </header>



        <div className={`${contentAnimClass} h-full`}>

          <div className="overflow-hidden h-full flex flex-col">

            {/* SUBTITLE/HELP BOX */}

            {(subtitle || helpText) && (

              <div className={`bg-white border-b-[5px] border-black px-6 py-3 transition-all flex items-center justify-between shrink-0`}>

                <div className="text-[11px] font-black uppercase tracking-widest text-black/70">

                  {subtitle}

                </div>

              </div>

            )}

            

            {/* MAIN CONTENT */}

            {(!unmountWhenClosed || open) && (

              <main className={`bg-white w-full h-full flex-1 ${finalContentClass}`}>{children}</main>

            )}

          </div>

        </div>

      </div>

    </div>

  );

};

// --- CUSTOM NEO-DROPDOWN COMPONENT (Shadows Removed) ---

interface NeoDropdownProps {

  options: string[];

  value: string;

  onChange: (val: string) => void;

  label?: string;

  leftColor?: string;

  rightColor?: string;

}



const NeoDropdown: React.FC<NeoDropdownProps> = ({ 

  options, 

  value, 

  onChange, 

  label, 

  leftColor = "bg-white", 

  rightColor = "bg-[#FFF07C]"

}) => {

  const [open, setOpen] = useState(false);

  const ddRef = useRef<HTMLDivElement>(null);



  useEffect(() => {

    const handleClick = (e: MouseEvent) => {

      if (ddRef.current && !ddRef.current.contains(e.target as Node)) setOpen(false);

    };

    document.addEventListener('mousedown', handleClick);

    return () => document.removeEventListener('mousedown', handleClick);

  }, []);



  return (

    <div className="relative w-full text-black flex flex-col justify-end h-14" ref={ddRef}>

      {/* Floating label prevents pushing the grid out of alignment */}

      {label && (

        <label className="absolute bottom-[100%] mb-1 left-2 text-[10px] font-black uppercase text-black/40 tracking-[0.2em] whitespace-nowrap z-10 pointer-events-none">

          {label}

        </label>

      )}

      

      <div className="relative w-full h-full">

        {/* Zero Shadows */}

        <button

          type="button"

          onClick={() => setOpen(!open)}

          className={`w-full h-full border-[4px] border-black ${open ? 'rounded-t-2xl rounded-b-none' : 'rounded-2xl'} flex items-stretch overflow-hidden cursor-pointer transition-all z-20 relative bg-white hover:translate-y-0.5 active:translate-y-1`}

        >

          <div className={`flex-1 ${leftColor} flex items-center px-4`}>

            <span className="text-[24px] font-[900] uppercase tracking-tighter leading-none mt-0.5">{value}</span>

          </div>

          <div className={`h-full w-14 border-l-[4px] border-black ${rightColor} flex items-center justify-center shrink-0`}>

            <ChevronDown size={24} strokeWidth={4} className={`text-black transition-transform duration-300 ${open ? 'rotate-180' : ''}`} />

          </div>

        </button>



        {/* Drops OUTSIDE grid via absolute positioning and high z-index */}

        {open && (

          <div className={`absolute top-full left-0 w-full border-[4px] border-t-0 border-black rounded-b-2xl z-50 flex flex-col overflow-hidden ${leftColor}`}>

            {options.map((opt, i) => (

              <button

                key={opt}

                type="button"

                onClick={() => { onChange(opt); setOpen(false); }}

                className={`w-full text-left px-4 py-3 text-xl font-[900] uppercase tracking-tighter hover:bg-black hover:text-white transition-colors ${i > 0 ? 'border-t-[4px] border-black' : ''}`}

              >

                {opt}

              </button>

            ))}

          </div>

        )}

      </div>

    </div>

  );

};

export type ToolboxVariant = 'scaffold' | 'accordion' | 'header';

export type ToolboxIndicator = 'symbols' | 'plusminus' | 'none';



interface ToolboxProps {

  variant?: ToolboxVariant;

  title: string;

  subtitle?: string;

  icon?: React.ReactNode;

  headerColor: string;

  iconBoxColor?: string;

  textColor?: string;

  collapsible?: boolean;

  isOpen?: boolean;

  isOpenInitial?: boolean;

  onToggle?: () => void;

  unmountWhenClosed?: boolean;

  headerActions?: React.ReactNode;

  indicator?: ToolboxIndicator;

  outerClassName?: string;

  shellClassName?: string;

  contentClassName?: string;

  embedded?: boolean;

  children?: React.ReactNode;

  helpTitle?: string;

  helpText?: string;

}



// --- MAIN TOOLBOX COMPONENT ---

export const Toolbox: React.FC<ToolboxProps> = ({

  variant = 'scaffold',

  title,

  subtitle,

  icon,

  headerColor,

  iconBoxColor = 'bg-white',

  textColor = 'text-black',

  collapsible = false,

  isOpen,

  isOpenInitial = true,

  onToggle,

  unmountWhenClosed = true,

  headerActions,

  indicator = 'symbols',

  outerClassName = '',

  shellClassName = '',

  contentClassName = '',

  embedded = false,

  children,

  helpTitle,

  helpText,

}) => {

  const [internalOpen, setInternalOpen] = useState(isOpenInitial);

  const [helpOpen, setHelpOpen] = useState(false);

  const [subtitleOpen, setSubtitleOpen] = useState(true);

  const helpIdRef = useRef(`help-${Math.random().toString(36).slice(2)}`);



  useEffect(() => {

    const handler = (e: Event) => {

      const detail = (e as CustomEvent<string>).detail;

      if (detail !== helpIdRef.current) {

        setHelpOpen(false);

      }

    };

    window.addEventListener('toolbox_help_open', handler as EventListener);

    return () => window.removeEventListener('toolbox_help_open', handler as EventListener);

  }, []);



  const controlled = typeof isOpen === 'boolean';

  const open = controlled ? Boolean(isOpen) : internalOpen;



  useEffect(() => {

    if (!open) {

      setHelpOpen(false);

      setSubtitleOpen(true);

    }

  }, [open]);



  const setOpen = () => {

    if (onToggle) {

      onToggle();

      return;

    }

    if (!controlled) setInternalOpen((prev) => !prev);

  };



  const isCollapsible = collapsible || indicator === 'plusminus' || indicator === 'symbols';

  const headerStyle = { backgroundColor: headerColor };

  const iconStyle = { backgroundColor: iconBoxColor };



  const finalContentClass = useMemo(() => {

    if (contentClassName) return contentClassName;

    if (variant === 'accordion') return 'p-6 bg-white text-black';

    return embedded ? 'p-0' : 'p-8';

  }, [contentClassName, embedded, variant]);



  if (variant === 'header') return null;



  // Strict Shadows: Scaffolds cast shadows. Accordions DO NOT.

  const frameClass =

    variant === 'accordion'

      ? 'w-full border-[4px] border-black rounded-2xl bg-white overflow-hidden relative shadow-none'

      : 'w-full bg-white border-[6px] border-black rounded-2xl shadow-[12px_12px_0px_0px_rgba(0,0,0,0.5)] flex flex-col overflow-hidden relative';



  const headerBorderThickness = variant === 'accordion' ? 'border-b-[4px]' : 'border-b-[6px]';

  const closedMargin = variant === 'accordion' ? 'mb-[-4px]' : 'mb-[-6px]';

  

  // Permanent negative margin overlaps the bottom border perfectly when closed

  const headerClass =

    variant === 'accordion'

      ? `${headerColor} ${textColor} h-14 flex items-center justify-between select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open ? closedMargin : 'mb-0'}`

      : `${headerColor} ${textColor} h-[80px] flex items-center justify-between overflow-hidden select-none relative z-20 group ${headerBorderThickness} border-black transition-[margin] duration-500 ease-in-out ${!open && !subtitle && !helpText ? closedMargin : 'mb-0'}`;



  const contentAnimClass =

    variant === 'accordion'

      ? `grid transition-[grid-template-rows] duration-500 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`

      : `grid transition-[grid-template-rows] duration-700 ease-in-out ${open ? 'grid-rows-[1fr]' : 'grid-rows-[0fr]'}`;



  return (

    <div className={`w-full h-full flex flex-col ${shellClassName} ${outerClassName}`}>

      <div className={`${frameClass} h-full`}>

        <header className={headerClass} style={headerStyle}>

          <div

            onClick={isCollapsible ? setOpen : undefined}

            className={`flex items-center h-full flex-1 ${isCollapsible ? 'cursor-pointer' : ''}`}

          >

            <div

              className={`${variant === 'accordion' ? 'h-full aspect-square border-r-[4px]' : 'h-full w-[80px] border-r-[6px]'} flex items-center justify-center border-black flex-shrink-0`}

              style={iconStyle}

            >

              {icon}

            </div>



            <div className={`flex flex-col ${variant === 'accordion' ? 'pl-4 justify-center' : 'pl-8'} pointer-events-none select-none`}>

              {variant === 'accordion' ? (

                <h3 className="text-[32px] font-[900] uppercase tracking-tighter leading-none mt-0.5">{title}</h3>

              ) : (

                <h1 className="text-[50px] font-[1000] uppercase tracking-tighter leading-none mt-1 sm:text-[40px] md:text-[50px] truncate">{title}</h1>

              )}

            </div>

          </div>



          <div className={`flex items-center ${variant === 'accordion' ? 'gap-3 pr-4' : 'gap-4 pr-6'} h-full`}>

            {headerActions}

            {indicator === 'plusminus' && (

              <div onClick={setOpen} className="h-full px-2 flex items-center justify-center cursor-pointer">

                <span className="text-2xl font-black leading-none text-black">{open ? '−' : '+'}</span>

              </div>

            )}

            {indicator === 'symbols' && isCollapsible && (

              <div onClick={setOpen} className="flex items-center justify-center h-full">

                <AnimatedToggleIcon open={open} size={variant === 'accordion' ? 36 : 48} />

              </div>

            )}

          </div>

        </header>



        <div className={`${contentAnimClass} h-full`}>

          <div className="overflow-hidden h-full flex flex-col">

            {/* SUBTITLE/HELP BOX */}

            {(subtitle || helpText) && (

              <div className={`bg-white border-b-[5px] border-black px-6 py-3 transition-all flex items-center justify-between shrink-0`}>

                <div className="text-[11px] font-black uppercase tracking-widest text-black/70">

                  {subtitle}

                </div>

              </div>

            )}

            

            {/* MAIN CONTENT */}

            {(!unmountWhenClosed || open) && (

              <main className={`bg-white w-full h-full flex-1 ${finalContentClass}`}>{children}</main>

            )}

          </div>

        </div>

      </div>

    </div>

  );

};
